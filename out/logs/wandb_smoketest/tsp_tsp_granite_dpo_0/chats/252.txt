import numpy as np

def heuristics(distance_matrix):
    # First, find the total number of nodes in the graph
    num_nodes = np.size(distance_matrix, 0)
    
    # Create a zeros matrix of the same shape as distance_matrix
    edge_importance = np.zeros_like(distance_matrix)
    
    # Calculate edge betweenness centrality using the formula:
    # Betweenness = (σst - (as * at)) / (as * at * (as * at - s))
    # Where σst is the number of shortest paths from s to t,
    # as is the number of shortest paths from s to any node,
    # at is the number of shortest paths from t to any node,
    # and s, t are nodes connected by an edge in the matrix.
    
    for i in range(num_nodes):
        for j in range(num_nodes):
            if i!= j:  # Skip the diagonal
                path_counts = np.sum((distance_matrix[:, i] <= distance_matrix[:, j]) & (distance_matrix[j, :] <= distance_matrix[i, :]))
                edge_importance[i, j] = path_counts / (path_counts + (distance_matrix[i, :] * distance_matrix[j, :]).sum() - path_counts * distance_matrix[i, :].sum() * distance_matrix[j, :].sum())
    
    return edge_importance
