```python
def heuristics(distance_matrix):
    num_nodes = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix)

    for i in range(num_nodes):
        for j in range(i, num_nodes):  # Consider only unvisited nodes
            # Benefit: Distance from i to j
            benefit = distance_matrix[i, j]
            
            # Cost: Distance from i to start (assumed to be 0 for the start node)
            cost_from_start = np.inf if i == 0 else distance_matrix[i, 0]

            # Cost: Distance from j to end (assumed to be 0 for the end node)
            cost_to_end = np.inf if j == num_nodes - 1 else distance_matrix[j, num_nodes - 1]
            
            # Benefit-to-cost ratio
            ratio = benefit / (cost_from_start + cost_to_end)
            
            heuristics_matrix[i, j] = ratio

    return heuristics_matrix
```
