import numpy as np

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros_like(distance_matrix)

    # Check for negative cycles
    if np.any(distance_matrix < 0):
        raise ValueError("Distance matrix contains negative cycles.")

    for i in range(n):
        for j in range(n):
            if i!= j:
                # Triangle inequality
                min_dist_ij = np.min(distance_matrix[i, :])
                min_dist_ji = np.min(distance_matrix[j, :])
                heuristic_matrix[i, j] = distance_matrix[i, j] + min(min_dist_ij, min_dist_ji) - distance_matrix[i, j]

    # Higher density areas may have more edges, hence less desirable
    density = np.sum(distance_matrix!= 0) / (n * (n - 1))
    heuristic_matrix += density

    return heuristic_matrix
