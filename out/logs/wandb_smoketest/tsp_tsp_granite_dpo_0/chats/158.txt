    hub_distance_matrix = np.diag(hub_degrees)

    spoke_lengths = np.minimum(distance_matrix, hub_distance_matrix)

    # Assign a high heuristic score to the edges connecting to high-degree nodes
    heuristic_matrix = spoke_lengths.copy()
