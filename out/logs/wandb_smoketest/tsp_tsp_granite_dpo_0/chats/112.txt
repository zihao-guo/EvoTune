```python
def heuristics(distance_matrix):
    n = len(distance_matrix)
    heuristic_matrix = np.zeros((n, n))
    
    # Diagonal elements (shortest paths within a node)
    np.fill_diagonal(heuristic_matrix, np.diagonal(distance_matrix))
    
    # Off-diagonal elements (inter-node distances) with penalty for larger values
    penalty_factor = 0.1  # Adjust this parameter for emphasis on shorter paths
    heuristic_matrix[np.arange(n), np.arange(n)] = distance_matrix[np.arange(n), np.arange(n)]
    heuristic_matrix[np.triu_indices(n, k=1)] *= (1 - penalty_factor)
    
    return heuristic_matrix
```
