import numpy as np

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    # Initialize the importance matrix
    importance_matrix = np.zeros((n, n))
    
    # Calculate the importance of each edge
    for i in range(n):
        for j in range(i+1, n):
            # Sum the minimum distance along the shortest paths
            importance_matrix[i, j] = min(distance_matrix[i, k] + distance_matrix[k, j] for k in range(n))

    # Normalize the importance matrix
    importance_matrix = importance_matrix / np.sum(importance_matrix, axis=1)[:, None]

    return importance_matrix
