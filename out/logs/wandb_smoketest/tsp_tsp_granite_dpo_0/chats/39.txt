def heuristics(distance_matrix):

    max_sum = np.sum(distance_matrix)

    heuristic_matrix = np.full(distance_matrix.shape, max_sum)

    # Implement Held-Karp lower bound heuristic
    for i in range(n):
        for j in range(i + 1, n):
            for k in range(j + 1, n):

                heuristic_matrix[j, i] = heuristic_matrix[i, j]  # symmetry
    
    return heuristic_matrix
