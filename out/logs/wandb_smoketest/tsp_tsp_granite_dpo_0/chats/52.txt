import numpy as np

def heuristics(distance_matrix):
    n = len(distance_matrix)  # assuming the matrix represents a graph with 'n' nodes

    # Initialize the heuristic matrix with the same shape as the distance matrix
    heuristics_matrix = np.zeros((n, n))

    # Calculate heuristic values for all edges
    for i in range(n):
        for j in range(n):
            # Opportunity cost = minimum distance from node j to any other unvisited node
            min_cost_from_j = np.min(distance_matrix[j, np.where(distance_matrix[j]!= 0)])
            
            # Current edge cost
            current_cost = distance_matrix[i, j]
            
            # Opportunity cost relative to current edge
            opportunity_cost_relative = (min_cost_from_j - current_cost) / min_cost_from_j
            
            # If current cost is smaller than opportunity cost, it's a good edge to keep
            if current_cost < opportunity_cost_relative:
                heuristics_matrix[i, j] = 0
            else:
                heuristics_matrix[i, j] = -current_cost  # Higher negative value indicates worse choice

    return heuristics_matrix
