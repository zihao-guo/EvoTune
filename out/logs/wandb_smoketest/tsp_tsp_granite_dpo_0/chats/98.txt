import numpy as np
import networkx as nx

def heuristics(distance_matrix):
    # Convert distance matrix to a graph
    graph = nx.from_numpy_array(distance_matrix)
    
    # Calculate Degree Centrality
    degree_centrality = nx.degree_centrality(graph)
    
    # Calculate Betweenness Centrality
    betweenness_centrality = nx.betweenness_centrality(graph)
    
    # Combine centrality measures into a single heuristics matrix
    heuristics_matrix = np.array(list(zip(degree_centrality.values(), betweenness_centrality.values())))
    
    return heuristics_matrix
