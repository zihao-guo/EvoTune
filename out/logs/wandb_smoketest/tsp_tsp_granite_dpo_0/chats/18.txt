def heuristics(distance_matrix):
    # Calculate the total distance for each node
    total_distances = np.sum(distance_matrix, axis=1)
    
    # Initialize the undesirability matrix with the same shape as the distance_matrix
    undesirability = np.copy(distance_matrix)
    
    # For each node, calculate the undesirability of all its edges
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            # If the current edge connects nodes with significant total distance difference,
            # assign a high undesirability score
            if i!= j and total_distances[i] > total_distances[j]:
                undesirability[i, j] = distance_matrix[i, j]
                undesirability[j, i] = distance_matrix[i, j]  # Reflect for the reversed edge
    
    return undesirability
