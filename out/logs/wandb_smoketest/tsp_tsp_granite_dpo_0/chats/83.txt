def heuristics(distance_matrix):
    # Get the shape of the distance matrix
    n, _ = distance_matrix.shape

    # Initialize a matrix of the same size with zeros
    heuristic_matrix = np.zeros_like(distance_matrix)

    # Calculate potential benefits
    for i in range(n):
        for j in range(i, n):  # Consider only lower-right triangle to avoid redundant calculations
            edge_distance = distance_matrix[i, j]

            potential_savings = min(distance_matrix[i, k] + distance_matrix[k, j] for k in range(n) if k!= i and k!= j) - edge_distance

            if potential_savings > 0:
