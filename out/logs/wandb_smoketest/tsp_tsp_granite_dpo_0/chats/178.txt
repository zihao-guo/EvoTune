import numpy as np

def heuristics(distance_matrix):

    out_degrees = np.zeros(n)

    node_probs = (betweenness / (degrees * (degrees - 1) / 2)) ** -1
    node_probs /= node_probs.sum()  # Normalize

    network = np.zeros((n, n))
