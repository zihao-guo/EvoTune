import numpy as np

def heuristics(distance_matrix):
    # Get the degrees of each node
    node_degrees = np.sum(distance_matrix, axis=0)
    
    # Calculate connectivity scores for each edge
    connectivity_scores = np.zeros_like(distance_matrix)
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            if i!= j:
                # Neighbors count for node i
                neighbors_count = np.sum(distance_matrix[i] < distance_matrix[j])
                connectivity_scores[i, j] = 1 / (neighbors_count + 1)

    # Combine degree and connectivity scores into a single heuristic matrix
    heuristics_matrix = np.maximum(node_degrees[:, np.newaxis], connectivity_scores)

    return heuristics_matrix
