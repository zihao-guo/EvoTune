import numpy as np

def heuristics(distance_matrix):
    # Calculate local connectivity density
    n = distance_matrix.shape[0]
    local_connectivity = np.zeros((n, n))
    for i in range(n):
        for j in range(i, n):
            neighbors = np.nonzero(distance_matrix[i, :] < distance_matrix[j, :] + 1)[0]
            if len(neighbors) > 1:
                local_connectivity[i, j] = len(neighbors) / (n - 1)
                local_connectivity[j, i] = local_connectivity[i, j]

    # Normalize distance matrix to make larger distances less desirable
    max_distance = np.max(distance_matrix)
    normalized_matrix = distance_matrix / max_distance

    # Combine factors: length, directionality, and local connectivity
    heuristic_matrix = np.zeros(distance_matrix.shape)
    heuristic_matrix += normalized_matrix
    heuristic_matrix *= distance_matrix
    heuristic_matrix += local_connectivity
    return heuristic_matrix
