In this approach, we calculate a relaxed distance between every pair of vertices, and then use the minimum of these distances as our heuristic score. 

```python
import numpy as np

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    # Initialize a distance matrix of the same shape for relaxed distances
    relaxed_distance_matrix = np.zeros_like(distance_matrix)

    # Calculate relaxed distances using the inverse of the original distance matrix
    for i in range(n):
        for j in range(n):
            if i!= j:
                relaxed_distance_matrix[i, j] = 1 / distance_matrix[i, j]

    # Calculate the minimum relaxed distance for each pair of vertices
    min_relaxed_distances = np.min(relaxed_distance_matrix, axis=0)

    # Create a heuristic score matrix with the same shape as the input
    heuristic_scores = np.minimum(distance_matrix, min_relaxed_distances)

    return heuristic_scores
```
