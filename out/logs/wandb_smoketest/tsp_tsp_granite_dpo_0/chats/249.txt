```python
import numpy as np

def heuristics(distance_matrix):
    # Calculate average degree of nodes
    avg_degree = np.mean(np.sum(distance_matrix, axis=1))

    # Create a new distance matrix where each edge's score is its weight minus the average degree
    undesirable_edges = distance_matrix - avg_degree

    # Ensure the resulting matrix is non-negative, as edge scores should not be negative
    undesirable_edges = np.maximum(undesirable_edges, 0)

    return undesirable_edges
```
