import numpy as np

def heuristics(distance_matrix):
    # Get the shape of the distance matrix
    n = distance_matrix.shape[0]

    # Initialize an array of zeros with the same shape as the distance matrix
    heuristics_matrix = np.zeros(distance_matrix.shape)

    # Iterate through each vertex
    for i in range(n):
        # Calculate the sum of the unvisited distances (current and its closest unvisited vertex)
        min_distance = float('inf')
        min_index = -1
        for j in range(n):
            if distance_matrix[i, j]!= 0:  # Avoid dividing by zero
                if j!= i:  # Ignore the edge between i and itself
                    if distance_matrix[i, j] < min_distance:
                        min_distance = distance_matrix[i, j]
                        min_index = j

        # If the vertex has unvisited neighbors, assign the minimum distance
        if min_index!= -1:
            heuristics_matrix[i, min_index] = min_distance

    return heuristics_matrix
