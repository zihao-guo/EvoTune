import numpy as np

def heuristics(distance_matrix):

    # Calculate the average distance
    avg_distance = np.mean(distance_matrix)
    
    # Density factor (1 for sparse, >1 for dense regions)
    density = 1 / np.count_nonzero(distance_matrix > 0)
    
    # Edge weight factor
    edge_weights_proportion = distance_matrix / np.linalg.norm(distance_matrix, axis=1)[:, np.newaxis]
    
    # Heuristic matrix
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    for i in range(num_nodes):
        for j in range(i+1, num_nodes):
            # Edge weight importance
            weight_importance = edge_weights_proportion[i, j]
            
            # Combine factors
            combined_factor = (avg_distance / (1 + np.abs(distance_matrix[i, j] - avg_distance))) * density * weight_importance
            
            # Introduce a softmax-like function to normalize heuristic values
            heuristic_matrix[i, j] = np.exp(-combined_factor) / np.sum(np.exp(-combined_factor))
            
    return heuristic_matrix
