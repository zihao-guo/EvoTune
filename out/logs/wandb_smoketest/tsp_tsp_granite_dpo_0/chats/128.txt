import numpy as np

def heuristics(distance_matrix):
    # Calculate the shortest path between every pair of nodes using Dijkstra's Algorithm
    # This provides a lower bound for the actual distance. Edges connected by the shortest path
    # should be less desirable due to the locality bias.
    shortest_paths = np.min(distance_matrix, axis=0) # This assumes the matrix is 2D for nodes and distances
    
    # Create a new matrix to hold heuristic scores
    heuristic_scores = np.copy(distance_matrix)

    # Apply locality-based penalty
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            # If the direct distance is less than or equal to the shortest path, apply penalty
            if distance_matrix[i, j] <= shortest_paths[i]:
                heuristic_scores[i, j] = -np.inf  # Treat this edge as undesirable
            else:
                heuristic_scores[i, j] = distance_matrix[i, j]

    return heuristic_scores
