def heuristics(distance_matrix):

    max_connectivity = (distance_matrix.shape[0] * (distance_matrix.shape[0] - 1)) // 2
    connectivity_score = connectivity / max_connectivity
    
    # 3. Compute a simple ranking based on the average of the sum of diagonal and non-diagonal elements
    average_score = (distance_matrix.sum() + distance_matrix.sum(axis=0).sum() + distance_matrix.sum(axis=1).sum()) / (3 * distance_matrix.shape[0])

    return heuristic_matrix
