```python
import numpy as np

def heuristics(distance_matrix):

    avg_distances = np.mean(distance_matrix, axis=0)
    
    # Create a new matrix where each element represents the weighted importance of the corresponding edge
    heuristics_matrix = np.maximum(avg_distances[:, np.newaxis], avg_distances[np.newaxis, :]) - distance_matrix
    
    return heuristics_matrix
```
