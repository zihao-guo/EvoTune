```python
def heuristics(distance_matrix):
    # Compute average distance for each node
    avg_distances = np.mean(distance_matrix, axis=0)
    
    # Create a new matrix with the same shape as the input
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Assign importance scores based on the average distance
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            heuristics_matrix[i][j] = 1 / avg_distances[i] + 1 / avg_distances[j]
    
    return heuristics_matrix
```
