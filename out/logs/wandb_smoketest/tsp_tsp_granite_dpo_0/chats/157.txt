```python
def heuristics(distance_matrix):
    num_vertices = distance_matrix.shape[0]
    adjacency_sum = np.sum(distance_matrix, axis=0)  # Sum of adjacent distances for each vertex
    total_distance = np.sum(distance_matrix)  # Total distance in the graph

    heuristics_matrix = np.zeros_like(distance_matrix)  # Initialize the result matrix with zeros

    for i in range(num_vertices):
        for j in range(num_vertices):
            if i!= j:  # Avoid self-loops
                heuristics_matrix[i][j] = adjacency_sum[j] + total_distance / (distance_matrix[i][j] * (num_vertices - 1))

    return heuristics_matrix
```
