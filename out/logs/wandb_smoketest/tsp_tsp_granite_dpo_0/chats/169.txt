```python
import numpy as np

def heuristics(distance_matrix):
    # Convert the distance matrix into a weighted graph
    graph = np.zeros((distance_matrix.shape[0], distance_matrix.shape[0]))
    for i in range(distance_matrix.shape[0]):
        for j in range(i, distance_matrix.shape[0]):
            graph[i][j] = distance_matrix[i][j]
            graph[j][i] = distance_matrix[i][j]  # Undirected graph

    # Calculate the degree centrality for each node
    degrees = np.sum(graph, axis=0)
    heuristic_matrix = np.diag(degrees)  # Diagonal matrix to ensure no self-loops

    return heuristic_matrix
```
