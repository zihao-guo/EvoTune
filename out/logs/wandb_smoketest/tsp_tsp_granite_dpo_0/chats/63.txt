def heuristics(distance_matrix):
    # Get dimensions of the matrix
    num_nodes = distance_matrix.shape[0]
    
    # Create a matrix of the same size filled with 1.0s
    heuristics_matrix = np.ones_like(distance_matrix)
    
    # Iterate over the matrix, setting heuristics based on inverse distance
    for i in range(num_nodes):
        for j in range(i+1, num_nodes):
            heuristics_matrix[i, j] = heuristics_matrix[j, i] = 1.0 / distance_matrix[i, j]
    
    return heuristics_matrix
