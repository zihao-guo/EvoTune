2. Connectivity Score: This score measures the number of other edges in the matrix that connect to an edge's endpoints. Higher connectivity implies the presence of an edge could enhance network coverage.
3. Randomness Factor: This is a random number between 0 and 1, providing some unpredictability and preventing over-reliance on calculated scores.

import numpy as np

def heuristics(distance_matrix):
    # Number of nodes
    num_nodes = distance_matrix.shape[0]
    proximity_scores = np.zeros((num_nodes, num_nodes))

    # Calculate proximity scores
    for i in range(num_nodes):
        for j in range(i + 1, num_nodes):
            proximity_scores[i, j] = distance_matrix[i, j]
            proximity_scores[j, i] = proximity_scores[i, j]

    # Calculate connectivity scores
    connectivity_scores = np.sum(distance_matrix > 0, axis=0)
    connectivity_scores = np.sum(distance_matrix > 0, axis=1).reshape(num_nodes, 1)

    # Randomness factor
    random_factors = np.random.rand(num_nodes, num_nodes)

    # Combine proximity, connectivity, and randomness into a single score
    combined_scores = proximity_scores + connectivity_scores + random_factors

    # Normalize the scores
    normalized_scores = np.minimum(combined_scores, 1)
    return normalized_scores
