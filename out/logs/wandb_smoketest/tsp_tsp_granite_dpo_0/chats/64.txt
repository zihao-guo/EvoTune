```python
def heuristics(distance_matrix):
    n = len(distance_matrix)
    heuristic_matrix = np.zeros_like(distance_matrix)

    for i in range(n):
        for j in range(n):
            # Skip self-loops
            if i == j:
                heuristic_matrix[i, j] = 0
            else:
                heuristic_matrix[i, j] = 1 / (distance_matrix[i, j] + 1)

    return heuristic_matrix
```
