```python
def heuristics(distance_matrix):
    # Calculate Manhattan Distance
    # Manhattan Distance for each pair of points (i, j) is sum(|i-j|, |i-k|, |j-k|)
    # For our case, we assume i, j, k are the indices of the points in the matrix
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((n, n))
    
    for i in range(n):
        for j in range(n):
            heuristic_matrix[i][j] = np.sum(np.abs(np.arange(n) - i)) + np.sum(np.abs(np.arange(n) - j))
            
    return heuristic_matrix
```
