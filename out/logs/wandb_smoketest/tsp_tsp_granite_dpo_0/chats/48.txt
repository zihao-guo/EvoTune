```python
import numpy as np

def heuristics(distance_matrix):
    if np.any(distance_matrix == 0):
        raise ValueError("Distance matrix should only contain positive values.")
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros((n, n))
    
    # Calculate harmonic mean for each pair of nodes
    for i in range(n):
        for j in range(n):
            if distance_matrix[i, j] > 0:  # Avoid division by zero
                heuristics_matrix[i, j] = n / (n * distance_matrix[i, j] / np.sum(distance_matrix[i, :]))
                heuristics_matrix[j, i] = heuristics_matrix[i, j]  # Symmetric heuristics matrix
    
    return heuristics_matrix

# You'll need to run the function with a distance matrix to see the results.
# For example:
distance_matrix = np.array([[0, 10, 15, 20],
                            [10, 0, 35, 25],
                            [15, 35, 0, 30],
                            [20, 25, 30, 0]])

print(heuristics(distance_matrix))
```

This function calculates the harmonic mean for each pair of nodes in the distance matrix.
