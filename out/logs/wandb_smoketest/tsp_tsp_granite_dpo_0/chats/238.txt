    manhattan = np.abs(distance_matrix - np.triu(distance_matrix, k=1))

    max_distance = np.max(euclidean)

    max_distance_reciprocal = 1 / max_distance

    heuristic_matrix = np.zeros(distance_matrix.shape)
    
    # Apply the Manhattan Distance
    heuristic_matrix += manhattan
    
    # Apply the Euclidean Distance

    # Apply the Reciprocal of the Maximum Distance
    heuristic_matrix += max_distance_reciprocal
    
    return heuristic_matrix
