```python
import numpy as np
from scipy.sparse.csgraph import shortest_path

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    degrees = np.sum(distance_matrix, axis=1)  # Degree centrality: sum of distances from a node
    in_degrees = np.minimum(degrees, np.sum(distance_matrix, axis=0))  # In-degree centrality: minimum of degrees from all nodes
    shortest_paths = shortest_path(distance_matrix, directed=False)  # Shortest path lengths

    heuristics_matrix = np.zeros_like(distance_matrix)

    # Apply heuristics
    for i in range(n):
        heuristics_matrix[i, :] = in_degrees[i] + np.sum(shortest_paths[i])  # Combine degree and shortest path

    return heuristics_matrix
```

It also uses the concept of shortest path lengths to prioritize nodes that are closer to most other nodes.
