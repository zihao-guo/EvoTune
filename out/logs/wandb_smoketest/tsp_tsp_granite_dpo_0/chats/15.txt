import numpy as np

def heuristics(distance_matrix):
    # Ensure distance_matrix is a numpy array
    distance_matrix = np.array(distance_matrix)

    # Compute connectivity scores
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((n, n))

    for i in range(n):
        for j in range(i+1, n):  # Ensure no self-connection
            connections_i = np.where(distance_matrix[i, :]!= np.inf)[0]
            connections_j = np.where(distance_matrix[j, :]!= np.inf)[0]
            common_connections = np.intersect1d(connections_i, connections_j)

            # Compute average distance to common connections
            avg_distance_i = np.mean(distance_matrix[i][common_connections])
            avg_distance_j = np.mean(distance_matrix[j][common_connections])
            avg_distance = (avg_distance_i + avg_distance_j) / 2

            heuristic_matrix[i, j] = avg_distance
            heuristic_matrix[j, i] = avg_distance  # For symmetry

    return heuristic_matrix
