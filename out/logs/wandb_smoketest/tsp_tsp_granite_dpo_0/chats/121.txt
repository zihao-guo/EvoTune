import numpy as np

def heuristics(distance_matrix):
    # Calculate the degree (connectivity) for each node
    node_degrees = np.sum(distance_matrix, axis=0)  # Degree for each row (node)
    
    # Calculate the inverse degree matrix (higher inverse for lower degrees)
    inverse_degrees = 1 / node_degrees
    
    # Inverse the degree matrix for the new heuristics values
    heuristics_matrix = -inverse_degrees[:, np.newaxis] * distance_matrix  # Multiply row-wise
    
    return heuristics_matrix
