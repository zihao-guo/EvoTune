```python
import numpy as np

def heuristics(distance_matrix):
    # Calculate the degree (sum of edge weights) for each node
    node_degrees = np.sum(distance_matrix, axis=0)
    
    # Calculate the total distance between all pairs of nodes
    total_distance = np.sum(distance_matrix)
    
    # Initialize the heuristic matrix
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    # Calculate the negatively weighted connectivity for each edge
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            if i!= j:
                # Calculate how much removing the edge (i,j) would reduce connectivity
                potential_disruption = (total_distance - distance_matrix[i, j]) / (node_degrees[i] * node_degrees[j])
                
                # Assign a high heuristic value to edges that could disrupt connectivity significantly
                heuristic_matrix[i, j] = (1 / potential_disruption) if potential_disruption > 1 else 0  # Avoid division by zero
                
    return heuristic_matrix
```
