def heuristics(distance_matrix):
    n_nodes = distance_matrix.shape[0]
    total_edges = int(distance_matrix.sum())  # Total number of edges

    # Initialize heuristic matrix with zeros
    heuristic_matrix = np.zeros(distance_matrix.shape)

    # Calculate edge density for each node
    for i in range(n_nodes):
        for j in range(i, n_nodes):  # Only consider edges between i and j
            if distance_matrix[i, j]!= 0:  # Ignore non-existent edges
                heuristic_matrix[i, j] = (2 * distance_matrix[i, j]) / total_edges
                heuristic_matrix[j, i] = heuristic_matrix[i, j]  # Symmetric

    return heuristic_matrix
