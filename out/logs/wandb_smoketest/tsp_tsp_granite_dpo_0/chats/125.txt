```python
import numpy as np

def heuristics(distance_matrix, importance_matrix=None):
    # If importance matrix is not provided, we'll assume all edges have equal importance
    if importance_matrix is None:
        importance_matrix = np.ones_like(distance_matrix)
    
    # Penalty: Sum of inverse distances
    penalty = np.sum(1 / distance_matrix, axis=1)
    
    # Importance indicator: Dot product of importance vectors for the edge's endpoints
    similarity = np.dot(importance_matrix[np.arange(len(distance_matrix)), :],
                         importance_matrix[:, np.arange(len(distance_matrix))])
    
    # Combine penalties and importance indicators
    undesirability = penalty * similarity
    
    return undesirability
```
