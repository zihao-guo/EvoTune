```python
def heuristics(distance_matrix, learning_rate=0.9):

    heuristic_matrix = distance_matrix.copy()

    # For each pair of vertices (i, j)
    for i in range(len(distance_matrix)):
        for j in range(len(distance_matrix[0])):
            if i!= j:  # Ignore diagonal elements
                # Calculate the Manhattan distance
                manhattan_dist = abs(distance_matrix[i][j]) + abs(distance_matrix[j][i])

                # Apply learning factor
                adjusted_dist = manhattan_dist * (1 - learning_rate)

                # Add this adjusted distance to the heuristic matrix
                heuristic_matrix[i][j] = adjusted_dist

    return heuristic_matrix
```
