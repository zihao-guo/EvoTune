```python
import numpy as np

def heuristics(distance_matrix):
    """
    This function assigns weights to each edge in the distance_matrix,
    using a weighted shortest path approach. Edges with larger distances
    are assigned smaller weights, making them less desirable.

    Args:
    distance_matrix: A 2D numpy array where distance_matrix[i][j] represents
                    the distance between nodes i and j.

    Returns:
    A 2D numpy array of the same shape as the input, where each entry
    represents the heuristic weight for the corresponding edge in the
    distance_matrix.
    """
    # Check if the input is a 2D numpy array
    if not isinstance(distance_matrix, np.ndarray) or distance_matrix.ndim!= 2:
        raise ValueError("Input must be a 2D numpy array.")
    
    # Calculate the number of nodes
    n_nodes = distance_matrix.shape[0]

    # Calculate the weight for each edge
    weights = np.zeros_like(distance_matrix)
    for i in range(n_nodes):
        for j in range(n_nodes):
            if i!= j:  # Avoid self-loops
                # Use the formula for the WSP heuristic: sqrt(distance)
                weights[i][j] = np.sqrt(distance_matrix[i][j])

    return weights
```

This function calculates a more sophisticated heuristic weight for each edge based on the square root of its distance.
