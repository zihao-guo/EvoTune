In this implementation, a lower score means the edge is less desirable to include in the solution. We'll negate the distances in the distance matrix before applying this heuristics function.

import numpy as np

def heuristics(distance_matrix):
    # Negate distances to turn higher values into lower ones
    neg_distance_matrix = -distance_matrix
    # Get the unique vertices
    n = neg_distance_matrix.shape[0]
    # Initialize the result matrix with zeros
    result = np.zeros_like(neg_distance_matrix)
    
    # Kruskal's Algorithm
    tree = []
    visited = set()
    edge_weights = list(zip(np.arange(n), np.arange(n)))
    
    while len(tree) < n - 1:
        # Find the smallest edge
        min_weight = float('inf')
        min_edge = None
        for i, j in edge_weights:
            if i not in visited and j not in visited:
                if neg_distance_matrix[i, j] < min_weight:
                    min_weight = neg_distance_matrix[i, j]
                    min_edge = (i, j)
        # Include the smallest edge in the tree
        tree.append(min_edge)
        visited.add(min_edge[0])
        visited.add(min_edge[1])
        
        # Check if adding this edge forms a cycle
        for k, l in edge_weights:
            if k in visited and l in visited and min_edge[0]!= k and min_edge[1]!= l:
                continue
            if neg_distance_matrix[k, l] < min_weight:
                edge_weights.remove((k, l))

    # Fill the result matrix with the weights of the edges in the tree
    for i, j in tree:
        result[i, j] = neg_distance_matrix[i, j]
    
    return result
