import numpy as np

def heuristics(distance_matrix):

    node_density = np.sum(distance_matrix, axis=1) / (distance_matrix.shape[0] * (distance_matrix.shape[0] - 1))

    # Initialize result matrix with the same shape as the input
    result_matrix = np.zeros_like(distance_matrix)

    # Fill the result matrix based on distance and density
    for i in range(distance_matrix.shape[0]):
        for j in range(i+1, distance_matrix.shape[0]):
            # Avoid diagonal elements
            if i!= j:
                result_matrix[i, j] = distance_matrix[i, j]
                result_matrix[j, i] = distance_matrix[i, j]  # Since distances are symmetric

    # Increase undesirability based on density
    result_matrix += (1 - node_density) * 100  # Adjust the factor for optimal balance

    return result_matrix
