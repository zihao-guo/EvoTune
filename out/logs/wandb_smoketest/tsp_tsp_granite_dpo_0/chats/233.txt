def heuristics(distance_matrix):

    # Calculate an edge's importance based on its position

            dist = distance_matrix[i, j]  # This is already provided in the input
            heuristic_matrix[i, j] = dist
            heuristic_matrix[j, i] = dist  # To avoid over-counting, also assign the same importance to reverse edges

    # Weights can be adjusted based on the specific characteristics of the problem
    # Here, we assign higher weights to edges connecting nodes with more neighbors
    weights = np.sum(distance_matrix, axis=1) / num_nodes  # Node i's total weighted distance to all other nodes
    heuristic_matrix *= weights

    return heuristic_matrix
