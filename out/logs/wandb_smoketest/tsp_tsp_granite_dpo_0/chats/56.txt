```python
def heuristics(distance_matrix):

    total_distance_per_row = np.sum(distance_matrix, axis=1)
    
    # Create a new matrix where each entry represents the importance of an edge
    # The importance of an edge (i, j) is the product of (1/total_distance_per_row[i]) and (1/total_distance_per_row[j])
    edge_importance = (1 / total_distance_per_row[:, np.newaxis]) * (1 / total_distance_per_row[np.newaxis, :])
    
    return edge_importance
```
