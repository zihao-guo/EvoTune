```python
import numpy as np

def heuristics(distance_matrix):
    n = len(distance_matrix)
    
    # Calculate local density
    local_density = np.sum(distance_matrix == min(distance_matrix.min(), distance_matrix.max()), axis=0) / (n * n - n)
    
    # Initialize heuristic matrix with zeros
    heuristic_matrix = np.zeros((n, n))
    
    # Populate the heuristic matrix
    for i in range(n):
        for j in range(i, n):
            # Edge connectivity
            connectivity = 1 / distance_matrix[i, j]
            
            # Edge weight
            weight = distance_matrix[i, j]
            
            # Combining connectivity and weight for local part
            local_part = connectivity + weight
            
            # Combining local density for global part
            global_part = local_density[i] + local_density[j]
            
            # Calculate the heuristic score
            heuristic_matrix[i, j] = local_part * global_part
            
            # Symmetry: heuristic(i, j) = heuristic(j, i)
            heuristic_matrix[j, i] = heuristic_matrix[i, j]
    
    return heuristic_matrix
```
