import numpy as np

def heuristics(distance_matrix):
    # Calculate the total distance for each vertex using LCV method
    n = distance_matrix.shape[0]
    total_distances = np.sum(distance_matrix, axis=1)
    
    # Compute the LCV heuristic score for each edge
    # LCV heuristic assigns a score based on the least constraining value, i.e., the smallest edge weight that remains once a vertex is chosen.
    heuristic_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i!= j:  # avoid self-loops
                # Exclude the chosen vertex for the current vertex
                remaining_weights = total_distances - distance_matrix[i, j]
                heuristic_matrix[i, j] = np.min(remaining_weights)
    
    # Ensure the diagonal elements are 0, as no edge should be included between a vertex and itself
    np.fill_diagonal(heuristic_matrix, 0)
    
    return heuristic_matrix
