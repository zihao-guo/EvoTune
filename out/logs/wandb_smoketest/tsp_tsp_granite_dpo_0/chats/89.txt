```python
def heuristics(distance_matrix):
    # Ensure the matrix is square
    n = distance_matrix.shape[0]

        raise ValueError("Input matrix must be square.")
    
    # Add a column of ones (flow conservation constraint)
    distance_matrix = np.c_[distance_matrix, np.ones(n)]
    
    # Initialize the flow heuristic matrix
    flow_heuristic = np.zeros((n, n))
    
    # Iterate over each node
    for i in range(n):
        # Calculate the total incoming flow for this node
        total_incoming_flow = np.sum(distance_matrix[i, :])
        
        # If total incoming flow is positive, this node is a source
        if total_incoming_flow > 0:
            # Assign a large negative value for undesirability to sources
            flow_heuristic[i, i] = -total_incoming_flow
        else:
            # For other nodes, calculate the sum of distances to all sources
            min_distance_to_source = np.min([distance_matrix[j, i] for j in range(n) if distance_matrix[j, i] > 0])
            
            # Assign a value proportional to the minimum distance to sources
            flow_heuristic[i, :] = min_distance_to_source
            flow_heuristic[:, i] = min_distance_to_source
    
    return flow_heuristic
```

This allows for a more nuanced evaluation of the undesirability of each edge in a potential solution.
