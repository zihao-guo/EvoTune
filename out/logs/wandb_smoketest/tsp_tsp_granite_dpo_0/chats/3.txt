def heuristics(distance_matrix):

    # Initialize heuristic scores with maximum possible values
    heuristic_scores = np.full_like(distance_matrix, np.inf)

    for i in range(n):

        remaining_unvisited_sum = 0
        for j in range(n):
            if j!= i:  # Skip the vertex itself
                remaining_unvisited_sum += distance_matrix[i, j]
        
        # Calculate the average distance to the remaining vertices
        avg_distance_to_remaining = remaining_unvisited_sum / (n - 1)
