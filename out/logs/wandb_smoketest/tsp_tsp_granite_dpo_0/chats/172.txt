import numpy as np

def heuristics(distance_matrix):
    # Calculate total weight for each row (nodes)
    total_weights = np.sum(distance_matrix, axis=1)

    # Calculate importance factor (square root of total weights)
    importance_factors = np.sqrt(total_weights)

    # Initialize the heuristics matrix with the same shape as the distance matrix
    heuristics_matrix = np.zeros_like(distance_matrix)

    # Fill the heuristics matrix
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            if i == j:  # Self-loops have zero importance
                heuristics_matrix[i][j] = 0
            else:
                # The heuristic value for an edge is the product of the direct distance and the importance factor of its two neighbors
                heuristics_matrix[i][j] = distance_matrix[i][j] * importance_factors[i] * importance_factors[j]

    return heuristics_matrix
