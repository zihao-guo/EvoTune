```python
def heuristics(distance_matrix):
    num_nodes = distance_matrix.shape[0]
    distance_sum = sum(sum(distance_matrix))
    
    # Calculate the mean distance between nodes
    mean_distance = distance_sum / (num_nodes * (num_nodes - 1))
    
    # Create a new matrix of the same shape as the input
    result_matrix = distance_matrix.copy()
    
    # Assign higher scores to edges connecting more distant nodes
    for i in range(num_nodes):
        for j in range(i+1, num_nodes):
            current_distance = distance_matrix[i][j]
            new_score = max(mean_distance - current_distance, 0)
            result_matrix[i][j] = new_score
            result_matrix[j][i] = new_score  # Symmetry in the problem, so edge (j, i) is also scored
    
    return result_matrix
```
