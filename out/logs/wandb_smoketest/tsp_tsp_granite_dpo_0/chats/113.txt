```python
def heuristics(distance_matrix):
    num_vertices = len(distance_matrix)
    heuristic_matrix = distance_matrix.copy()  # Initialize as a copy of the distance matrix

    for vertex in range(num_vertices):
        # Find the index of the farthest unvisited vertex
        farthest_index = max(range(num_vertices), key=lambda i: heuristic_matrix[i][vertex])
        
        # Insert the farthest vertex at the current position
        heuristic_matrix[vertex][farthest_index] = distance_matrix[vertex][farthest_index]

    return heuristic_matrix
```
