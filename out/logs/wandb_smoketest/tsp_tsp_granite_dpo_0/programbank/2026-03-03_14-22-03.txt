    - distance_matrix: (N, N) numpy array.
    - path_count: (N, N) numpy array, optional. Initial path frequency count. Default is None.

    - time_decay: float. Decay factor for path frequencies. Default is 0.95.

    # Calculate new path count (weighted by time decay)

                                 np.where(distance_matrix > 0, 
                                              path_count * time_decay,

                                 0)

    # High-flow edges

    high_flow_edges = np.sum(high_flow_mask * path_count, axis=1) > high_flow_threshold
    heuristic_matrix[high_flow_edges] = distance_matrix[high_flow_edges] * 1.5  # Scale up high-flow edges

    return heuristic_matrix

import numpy as np
from scipy.sparse.csgraph import dijkstra
def heuristics(distance_matrix):
    # Create a (n x n) adjacency matrix based on the given distance matrix
    n = distance_matrix.shape[0]
    adj_matrix = np.zeros((n, n))
    np.fill_diagonal(adj_matrix, 1)  # Each node is connected to itself with a weight of 1
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j]!= float('inf'):  # Skip edges with infinite distance
                adj_matrix[i, j] = distance_matrix[i, j]
                adj_matrix[j, i] = distance_matrix[i, j]  # Make it a bidirectional edge

    # Calculate shortest path from each node to all others using Dijkstra's algorithm
    shortest_paths = dijkstra(csgraph=adj_matrix, directed=True, return_predecessors=False)
    
    # Normalize by the number of shortest paths for each edge
    normalized_heuristics = shortest_paths / np.sum(shortest_paths, axis=1)[:, np.newaxis]
    
    # Combine distance and importance by multiplying them together
    heuristics_matrix = distance_matrix * normalized_heuristics
    
    return heuristics_matrix
