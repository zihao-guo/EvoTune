import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    degrees = np.sum(distance_matrix, axis=0)

    weights = distance_matrix.copy()
    
    # Normalize weights by node degrees
    weights /= degrees[:, np.newaxis]

    mst_weights, _ = minimum_spanning_tree(csgraph=weights, directed=False)

    neighbor_matrix = distance_matrix.copy()

    edge_counts = np.sum(distance_matrix, axis=0)
    avg_edge_count = edge_counts / np.sum(edge_counts)
