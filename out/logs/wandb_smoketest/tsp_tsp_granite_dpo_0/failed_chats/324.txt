```python
import numpy as np
from scipy.linalg import eigs

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    # Create the adjacency matrix from distance matrix
    adj_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i!= j:
                adj_matrix[i, j] = 1 - distance_matrix[i, j] # Assuming 0 as connection
    # Calculate the eigenvectors of the adjacency matrix
    eigenvalues, eigenvectors = eigs(adj_matrix, k=1)
    # Compute centrality scores
    centrality_scores = np.abs(eigenvectors) / np.sum(eigenvectors)
    # Normalize scores to sum to 1
    centrality_scores /= np.sum(centrality_scores)
    
    # Return a matrix of undesirability scores
    return np.array([1 - centrality_scores[i, j] for i in range(n) for j in range(n)])
```
