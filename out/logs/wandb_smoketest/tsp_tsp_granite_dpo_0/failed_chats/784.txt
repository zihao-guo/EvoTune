```python
import numpy as np

def heuristics(distance_matrix):
    # Convert the distance matrix into a graph
    n = distance_matrix.shape[0]
    graph = np.zeros((n, n))

    for i in range(n):
        for j in range(i, n):
            graph[i][j] = graph[j][i] = distance_matrix[i][j]

    # Weighted Shortest Path Heuristic
    def wsp(start, end):
        visited = np.zeros(n, dtype=bool)
        path_length = np.full(n, np.inf)
        path_length[start] = 0

        while np.any(visited!= np.union1d(visited, np.where(path_length!= np.inf)[0])):
            min_length_vertex = np.argmin(path_length)
            visited[min_length_vertex] = True

            for neighbor in range(n):
                if graph[min_length_vertex][neighbor]!= np.inf and not visited[neighbor]:
                    new_length = path_length[min_length_vertex] + graph[min_length_vertex][neighbor]
                    if new_length < path_length[neighbor]:
                        path_length[neighbor] = new_length

    # Compute and store the number of edges in the shortest path
    heuristic_scores = np.zeros_like(distance_matrix)
    for i in range(n):
        wsp(i, i)  # path length is 0 for a vertex
        for j in range(i+1, n):
            wsp(i, j)
            heuristic_scores[i][j] = path_length[j] - 1  # subtract 1 to get number of edges in shortest path

    return heuristic_scores
```

Then, for each pair of vertices, it computes the shortest path (using a simple iterative approach) and counts the number of edges in this path.
