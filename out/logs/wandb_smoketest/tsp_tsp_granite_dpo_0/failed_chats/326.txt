def wdh_heuristics(distance_matrix, weights=None):
    """
    Calculate a heuristic score for each entry in the distance_matrix based on weighted distances.
    
    Args:
    - distance_matrix (list of list): A 2D list representing the distance matrix.
    - weights (list, optional): A list of weights corresponding to each row in the distance matrix.
                                If not provided, all weights are assumed to be equal.
    
    Returns:
    - list of list: A 2D list with the same shape as distance_matrix, containing the WDH scores.
    """
    if weights is None:
        # Assuming all weights are equal
        weights = [1 for _ in range(len(distance_matrix))]

    num_rows, num_cols = len(distance_matrix), len(distance_matrix[0])
    wdh_matrix = [[0 for _ in range(num_cols)] for _ in range(num_rows)]

    # Calculate the weighted distance for each entry
    for i in range(num_rows):
        for j in range(num_cols):
            wdh_matrix[i][j] = distance_matrix[i][j] * weights[i]

    return wdh_matrix
