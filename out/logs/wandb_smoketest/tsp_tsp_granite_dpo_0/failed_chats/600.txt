def heuristics(distance_matrix):

                heapq.heappush(priority_queue, (distances[neighbor], neighbor))
    
    # Calculate heuristic values as the difference between total cost to node and cost to neighbor
    heuristic_values = [[distance_matrix[i][j] - distances[i] for j in range(n)] for i in range(n)]
