import numpy as np

def heuristics(distance_matrix):

    def connection_weight(node, connections):
        return 1 / (np.sum(np.abs(node - connections)) + 1)

    node_connections = np.argsort(distance_matrix, axis=1)[:,1:]

    weight_matrix = np.zeros(distance_matrix.shape)
    for i in range(distance_matrix.shape[0]):

            weight_matrix[i,j] = connection_weight(i, j)
    
    # Calculate the "distance-based cost" for each edge
    # This is a simple form that increases with distance and decreases with weight
    distance_based_cost = distance_matrix + weight_matrix
    
    # Normalize to ensure it's a valid score matrix

    if max_cost == 0:
        return distance_based_cost
    else:
        return distance_based_cost / max_cost
