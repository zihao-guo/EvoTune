```python
import numpy as np

def weighted_heuristics(distance_matrix):
    # Calculate the cost for each edge using the inverse of the distance
    # This means, shorter distances have lower 'cost'
    edge_costs = 1 / distance_matrix  # Edge costs are reciprocal of distances

    # Normalize the costs so that they sum up to 1. This means each edge contributes equally to the total cost
    total_cost = np.sum(edge_costs)
    edge_costs /= total_cost

    return edge_costs
```
