```python
import heapq

def heuristics(distance_matrix):
    # Create a disjoint set data structure
    parent = list(range(len(distance_matrix)))

    def find(x):
        if parent[x]!= x:
            parent[x] = find(parent[x])
        return parent[x]

    def union(x, y):
        root_x = find(x)
        root_y = find(y)
        if root_x!= root_y:
            parent[root_x] = root_y

    # Kruskal's algorithm to maintain a set of edges forming an MST
    mst_edges = []
    edges_heap = [(distance_matrix[i][j], i, j) for i in range(len(distance_matrix)) for j in range(i+1, len(distance_matrix))]
    heapq.heapify(edges_heap)

    while edges_heap:
        cost, u, v = heapq.heappop(edges_heap)
        if find(u)!= find(v):  # Ensure they are not already connected
            union(u, v)
            mst_edges.append((u, v, cost))

    # Create the heuristics matrix
    heuristics_matrix = [[0] * len(distance_matrix) for _ in range(len(distance_matrix))]
    for u, v, _ in mst_edges:
        heuristics_matrix[u][v] = heuristics_matrix[v][u] = -distance_matrix[u][v]  # Necessary for symmetry

    return heuristics_matrix
```

This function works by constructing a Minimum Spanning Tree (MST) from the input distance matrix. The heuristic score for each edge is then the negative of its weight in the MST. The disjoint set data structure ensures that duplicate edges are not included in the MST, thus preventing over-counting of potential undesirable edges.
