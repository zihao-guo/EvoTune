import numpy as np

def heuristics(distance_matrix):
    # Initialize the result matrix
    num_nodes = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((num_nodes, num_nodes))

    # Adaptive edge weight decay
    decay_factor = 0.5  # adjust this value based on your needs
    for i in range(num_nodes):
        for j in range(num_nodes):
            if i == j:
                heuristic_matrix[i][j] = distance_matrix[i][j]
            else:
                distance_ij = distance_matrix[i][j]
                distance_goal = min(distance_matrix[i][goal] for goal in range(num_nodes) if goal!= i)
                distance_heuristic = distance_ij + decay_factor * (distance_goal - distance_ij)
                heuristic_matrix[i][j] = distance_heuristic

    # Adjacency-based global influence
    adjacent_edges_count = np.sum(distance_matrix, axis=1)  # count edges connected to each node
    heuristic_matrix += np.transpose(adjacent_edges_count)  # mirror for symmetric matrix

    # Normalize the heuristic matrix
    norm = np.sum(heuristic_matrix, axis=1)
    heuristic_matrix /= norm[:, np.newaxis]

    return heuristic_matrix
