def heuristics(distance_matrix):
    # Calculate the sum of each row, representing the total distance from that node
    row_sums = np.sum(distance_matrix, axis=1)
    
    # Calculate the total sum of all distances in the matrix
    total_sum = np.sum(distance_matrix)
    
    # Edge redundancy heuristic: for each edge (i, j), if the path through i also passes through some other node k (i.e., i < k < j), increase the heuristic score for this edge
    heuristic_matrix = np.zeros_like(distance_matrix)
    for i in range(len(distance_matrix)):
        for j in range(i+1, len(distance_matrix)):
            for k in range(i, len(distance_matrix)):
                if distance_matrix[i][k] + distance_matrix[k][j] > distance_matrix[i][j]:
                    heuristic_matrix[i][j] += row_sums[k] - total_sum
                elif distance_matrix[i][k] + distance_matrix[k][j] == distance_matrix[i][j]:
                    heuristic_matrix[i][j] += row_sums[k]
    
    return heuristic_matrix
