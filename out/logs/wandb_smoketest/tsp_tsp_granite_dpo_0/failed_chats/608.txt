import numpy as np

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    adjacency_matrix = np.abs(distance_matrix)  # Square of distances, assumed unweighted
    degree = np.sum(adjacency_matrix, axis=0)  # Degree of each node
    clustering_coefficients = np.zeros((n, n))

    for i in range(n):
        for j in range(n):
            if adjacency_matrix[i, j] > 0:
                triangles = np.sum(adjacency_matrix[i, :] * adjacency_matrix[:, j] * adjacency_matrix[np.outer(np.arange(n), np.arange(n)) == np.outer(i, j)])
