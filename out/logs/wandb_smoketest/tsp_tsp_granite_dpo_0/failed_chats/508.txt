def heuristics(distance_matrix):
    import numpy as np
    
    # Calculate the degree of each vertex (number of edges connected to it)
    vertex_degrees = np.sum(distance_matrix, axis=1)

    avg_distances = np.mean(distance_matrix, axis=1)
    
    # Create a heuristic matrix.
    # Each cell (i,j) contains a negative value if adding an edge between vertices i and j would be undesirable,
    # and a positive value if it would be desirable.
    heuristic_matrix = np.where(distance_matrix > 1, # Assuming cost > 1 for an undesirable edge
                               -vertex_degrees[np.arange(len(distance_matrix))][distance_matrix[np.arange(len(distance_matrix)), np.arange(len(distance_matrix)) > 1]] +
                               avg_distances[np.arange(len(distance_matrix))][distance_matrix[np.arange(len(distance_matrix)), np.arange(len(distance_matrix)) > 1]] - 1,  # Subtract 1 to make it a desirable score for undesirable edges
                               0)
    
    return heuristic_matrix

1. Degree: Higher-degree vertices have fewer connections, and adding an edge to them might disrupt their existing connections more.
2. Average Distance: A vertex that is already well-connected (low average distance) might not want to add more edges, as it would increase its overall path lengths.
