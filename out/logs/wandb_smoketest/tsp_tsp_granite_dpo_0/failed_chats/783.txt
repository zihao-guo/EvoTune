def floyd_warshall(distance_matrix):
    # Initialize the graph as the input distance matrix
    n = len(distance_matrix)
    graph = [[float('inf') for _ in range(n)] for _ in range(n)]

    # Each node is its own shortest path
    for i in range(n):
        graph[i][i] = 0

    # Fill in the graph with the shortest path values
    for k in range(n):
        for i in range(n):
            for j in range(n):
                graph[i][j] = min(graph[i][j], graph[i][k] + graph[k][j])

    return graph
