```python
def heuristics(distance_matrix):
    n = len(distance_matrix)
    # Union-find data structure to keep track of connected components
    parent = list(range(n))
    
    def find(x):
        if parent[x]!= x:
            parent[x] = find(parent[x])
        return parent[x]
    
    def union(x, y):
        root_x = find(x)
        root_y = find(y)
        
        if root_x == root_y:
            return False
        
        parent[root_x] = root_y
        return True
    
    # Kruskal's Algorithm to build Minimum Spanning Tree
    mst_edges = []
    edges_sorted = sorted((weight, i, j) for i, row in enumerate(distance_matrix) for j, weight in enumerate(row))
    
    for weight, u, v in edges_sorted:
        if find(u)!= find(v):
            mst_edges.append((weight, u, v))
            union(u, v)
    
    # Heuristic matrix where each entry is the weight of the corresponding edge in MST
    heuristic_matrix = [[0] * n for _ in range(n)]
    for weight, u, v in mst_edges:
        heuristic_matrix[u][v] = weight
    heuristic_matrix = [[-1 * weight for _ in range(n)] for weight in heuristic_matrix]
    
    return heuristic_matrix
```
