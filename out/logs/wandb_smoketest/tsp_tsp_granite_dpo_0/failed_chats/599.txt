def heuristics(distance_matrix):
    n = len(distance_matrix)
    # Degree of each node
    degrees = [sum(distance_matrix[i]) for i in range(n)]
    
    # Initialize heuristics matrix
    heuristics_matrix = [[0 for _ in range(n)] for _ in range(n)]
    
    # Calculate closeness centrality with degree-penalty
    for i in range(n):
        for j in range(n):
            if i == j:
                heuristics_matrix[i][j] = 0  # Self-edge penalty
            else:
                heuristics_matrix[i][j] = 1 / distance_matrix[i][j]
                heuristics_matrix[i][j] *= degrees[i] / degrees[j]  # Degree-penalty
    
    return heuristics_matrix
