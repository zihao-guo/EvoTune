def heuristics(distance_matrix):

    for i in range(n):
        nearest_idx = np.argmin(distance_matrix[i])
        heuristic_matrix[i, nearest_idx] = distance_matrix[i, nearest_idx]

    for i in range(n):
        for j in range(i+1, n):
            min_cost = float('inf')

                    if cost < min_cost:
                        min_cost = cost
            heuristic_matrix[i][j] = min_cost
            heuristic_matrix[j][i] = min_cost  # Symmetric 2-Opt

    # Add connectivity weight
    for i in range(n):
        for j in range(n):
            if i!= j:

                common_neighbors = np.count_nonzero(distance_matrix[i] <= distance_matrix[j])

                heuristic_matrix[i][j] += weight

    return heuristic_matrix
