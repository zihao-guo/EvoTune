import numpy as np

def shortest_path(distance_matrix, start_vertex):
    num_vertices = distance_matrix.shape[0]
    visited = np.zeros(num_vertices, dtype=bool)
    predecessor = np.empty(num_vertices, dtype=int)
    distance = np.Inf * np.ones(num_vertices, dtype=float)
    distance[start_vertex] = 0

    queue = [(0, start_vertex)]

    while queue:
        current_distance, current_vertex = queue.pop(0)

        if current_distance > distance[current_vertex]:
            continue

        for neighbor, edge_cost in enumerate(distance_matrix[current_vertex]):
            if visited[neighbor] or edge_cost == float('Inf'):
                continue

            new_distance = current_distance + edge_cost
            if new_distance < distance[neighbor]:
                distance[neighbor] = new_distance
                predecessor[neighbor] = current_vertex
                queue.append((new_distance, neighbor))
                visited[neighbor] = True

    return distance, predecessor

def heuristics(distance_matrix, start_vertex):
    distance, predecessor = shortest_path(distance_matrix, start_vertex)
    heuristics_matrix = np.empty(distance_matrix.shape, dtype=float)

    for vertex in range(distance_matrix.shape[0]):
        for neighbor in range(distance_matrix.shape[0]):
            if predecessor[neighbor] == vertex:
                heuristics_matrix[neighbor][vertex] = 1 / distance[neighbor]
                break
            elif distance[neighbor] < distance[vertex]:
                heuristics_matrix[neighbor][vertex] = 1 / distance[neighbor]

    return heuristics_matrix
