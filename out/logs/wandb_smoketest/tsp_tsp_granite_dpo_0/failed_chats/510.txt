def mst_kruskal(distance_matrix):

    edges = [(distance_matrix[i][j], i, j) for i in range(n) for j in range(i+1, n)]
    heapq.heapify(edges)
    parent = list(range(n))
    mst = [0] * n

        if parent[u]!= parent[v]:
            mst[u] = mst[v] = 1
            parent[v] = parent[u]
            for w in range(n):
                if parent[w] == parent[v]:
                    heapq.heappush(edges, (distance_matrix[v][w], v, w))

    return mst

def heuristics(distance_matrix):

    mst = mst_kruskal(distance_matrix)

    heuristic_matrix = np.zeros(distance_matrix.shape)
    for u in range(n):
