This approach considers not just the physical distance, but also the network's robustness and resilience.

```python
import numpy as np

def floyd_warshall(distance_matrix):
    num_nodes = distance_matrix.shape[0]
    for k in range(num_nodes):
        for i in range(num_nodes):
            for j in range(num_nodes):
                distance_matrix[i, j] = min(distance_matrix[i, j], distance_matrix[i, k] + distance_matrix[k, j])
    return distance_matrix

def heuristics(distance_matrix):
    # Floyd-Warshall to get effective resistance
    effective_resistance = floyd_warshall(distance_matrix)
    
    # Let's assume higher values are less desirable
    # Calculate weighted sum of effective resistance and distance
    heuristics_matrix = np.where(distance_matrix > effective_resistance, 1 / effective_resistance, distance_matrix)
    
    # Normalize to ensure the matrix values are between 0 and 1
    heuristics_matrix = np.maximum(heuristics_matrix, 0) / np.max(heuristics_matrix)
    
    return heuristics_matrix
```
