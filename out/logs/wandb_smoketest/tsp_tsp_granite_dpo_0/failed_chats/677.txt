```python
import numpy as np
from scipy.sparse.csgraph import csgraph_ shortest_paths

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]

    # Compute the shortest path matrix using scipy's csgraph
    shortest_path = csgraph_shortest_paths(csgraph=csgraph_from_dense(distance_matrix),
                                         method='dijkstra')

    # Calculate the edge betweenness centrality
    betweenness_centrality = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            path_length = shortest_path[i, j]
            num_paths = np.sum(np.isfinite(shortest_path[i, :])) * np.sum(np.isfinite(shortest_path[:, j]))
            if path_length < n:
                betweenness_centrality[i, j] = np.sum((shortest_path[i, :][np.argsort(shortest_path[i, :])] == j) & (shortest_path[:, j][np.argsort(shortest_path[:, j])] == i)) / (num_paths - 1)
            elif path_length == n:
                betweenness_centrality[i, j] = 0

    # Normalize betweenness centrality to achieve a matrix of values between 0 and 1
    betweenness_centrality = (betweenness_centrality - np.min(betweenness_centrality)) / (np.max(betweenness_centrality) - np.min(betweenness_centrality))

    return betweenness_centrality
```
