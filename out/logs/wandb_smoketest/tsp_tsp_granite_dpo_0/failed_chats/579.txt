def heuristics_improved(distance_matrix):
    n = len(distance_matrix)
    is_in_mst = [False] * n
    parent = [-1] * n
    rank = [0] * n
    mst = []

        if parent[x] == -1:
            return x
        return find(parent[x])

    def union(x, y):
        root_x = find(x)
        root_y = find(y)
        if root_x == root_y:
            return

        if rank[root_x] > rank[root_y]:
            parent[root_y] = root_x
        elif rank[root_x] < rank[root_y]:
            parent[root_x] = root_y
        else:
            parent[root_y] = root_x
            rank[root_x] += 1

    def kruskal_mst(edges):
        heapq.heapify(edges)
        nonlocal mst
        mst = []
        total_weight = 0
        while edges and total_weight < n - 1:
            edge = heapq.heappop(edges)
            u, v, weight = divmod(edge, n)
            if find(u)!= find(v):
                union(u, v)
                mst.append((u, v, weight))
                total_weight += weight
        return mst

    # Precompute edges sorted by distance
    edges = sorted((i, j, distance_matrix[i][j]) for i in range(n) for j in range(i+1, n) if i < j)
    edges_to_mst = []

    for u, v, weight in edges:
        if find(u)!= find(v):
            union(u, v)
            edges_to_mst.append((u, v, weight))
            is_in_mst[u] = True
            is_in_mst[v] = True

    # Create the penalty matrix
    penalty_matrix = [[0] * n for _ in range(n)]
    for u, v, _ in edges_to_mst:
        penalty_matrix[u][v] = distance_matrix[u][v]
        penalty_matrix[v][u] = distance_matrix[u][v]  # Symmetry for undirected graph

    return penalty_matrix
