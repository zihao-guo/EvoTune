```python
import numpy as np

def heuristics(distance_matrix):
    # Get the shape of the matrix
    m, n = distance_matrix.shape

    # Initialize a new matrix for heuristic values
    heuristic_matrix = np.zeros((m, n))

    # Calculate the normalized sum of distances for each edge
    edge_sums = np.sum(distance_matrix, axis=1)

    # Normalize by the maximum edge sum
    max_edge_sum = np.max(edge_sums)
    heuristic_matrix = edge_sums / max_edge_sum

    # Add a time-decay factor for future impact
    time_decay = 0.5  # You can adjust this value
    for i in range(m):
        for j in range(n):
            heuristic_matrix[i, j] *= time_decay ** np.abs(i - j)  # Distance impact
            heuristic_matrix[i, j] *= time_decay ** distance_matrix[i, j]  # Distance value

    return heuristic_matrix
```
