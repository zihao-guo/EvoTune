```python
def heuristics(distance_matrix):
    n = len(distance_matrix)
    # Initialize the heuristics matrix with the same values as the distance matrix
    heuristics_matrix = distance_matrix.copy()
    
    # Calculate improvement costs
    for k in range(n):
        for i in range(n):
            for j in range(n):
                if i!= k and j!= k:
                    # Compute the improvement cost for edge (i, j)
                    improvement_cost = distance_matrix[i, j] - distance_matrix[i, k] - distance_matrix[k, j] + distance_matrix[i, k] + distance_matrix[k, j]
                    heuristics_matrix[i, j] += improvement_cost
    
    return heuristics_matrix
```
