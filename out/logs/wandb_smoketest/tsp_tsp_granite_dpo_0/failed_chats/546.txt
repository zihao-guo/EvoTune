import numpy as np

def dijkstra(graph, start):
    """
    Function to perform Dijkstra's shortest path algorithm
    """
    N = len(graph)
    dist = np.inf * np.ones(N, dtype=np.float32)
    dist[start] = 0
    prev = np.empty(N, dtype=np.int32)
    queue = [(0, start)]
    
    while queue:
        d, u = queue.pop(0)
        if d!= dist[u]:
            continue
        for v, weight in enumerate(graph[u]):
            if weight and dist[v] > dist[u] + weight:
                dist[v] = dist[u] + weight
                prev[v] = u
                queue.append((dist[v], v))
    
    return dist, prev

def heuristics(distance_matrix):
    N = distance_matrix.shape[0]
    dist, prev = dijkstra(np.zeros((N, N), dtype=np.float32), 0)

    # Calculate the shortest path difference
    shortest_paths = np.zeros((N, N), dtype=np.float32)
    for i in range(N):
        for j in range(N):
            if dist[i][j]!= np.inf:
                shortest_paths[i][j] = dist[i][j]
                for k in range(N):
                    if prev[i] == k and prev[j] == k:
                        shortest_paths[i][j] += distance_matrix[i][k] * distance_matrix[k][j]

    # Compute the heuristic value
    heuristic_values = np.zeros((N, N), dtype=np.float32)
    for i in range(N):
        for j in range(N):
            heuristic_values[i][j] = distance_matrix[i][j] + shortest_paths[i][j] - dist[i][j]

    return heuristic_values
