```python
def heuristics(distance_matrix):
    num_nodes = distance_matrix.shape[0]
    heuristic_matrix = [[0.0 for _ in range(num_nodes)] for _ in range(num_nodes)]

    for i in range(num_nodes):
        for j in range(num_nodes):
            if i!= j:
                # Calculate the disruption caused by including the edge (i, j)
                disruption = distance_matrix[i][j] + min(distance_matrix[i][k] for k in range(num_nodes) if k!= i and k!= j)

                heuristic_matrix[i][j] = disruption + 1.0

    # Normalize heuristic values so they can be compared directly
    max_heuristic = max(map(max, heuristic_matrix))
    for i in range(num_nodes):
        for j in range(num_nodes):
            if i!= j:
                heuristic_matrix[i][j] /= max_heuristic

    return heuristic_matrix
```
