import heapq

def heuristics(distance_matrix):
    # Get the shape of the input matrix
    num_nodes = len(distance_matrix)
    
    # Initialize a dictionary to store tentative distances
    tentative_distances = {node: (distance_matrix[node][node], node) for node in range(num_nodes)}
    
    # Set of unvisited nodes
    unvisited = set(range(num_nodes))

    # Priority queue to store (distance, node)
    # Lower priority means a shorter path to this node
    priority_queue = [(0, node) for node in unvisited]
    heapq.heapify(priority_queue)

    # Dijkstra's algorithm
    while priority_queue:
        # Get the node with the lowest tentative distance
        current_distance, current_node = heapq.heappop(priority_queue)

        # Skip if we've already computed a better path for this node
        if current_distance > tentative_distances[current_node][0]:
            continue

        # Update tentative distances to neighboring nodes
        for neighbor, weight in enumerate(distance_matrix[current_node]):
            if neighbor not in unvisited:
                continue

            new_distance = current_distance + weight

            # If a shorter path to this neighbor is found
            if new_distance < tentative_distances[neighbor][0]:
                tentative_distances[neighbor] = (new_distance, neighbor)
                heapq.heappush(priority_queue, (new_distance, neighbor))

    # Prepare the heuristic matrix
    heuristic_matrix = [[0] * num_nodes for _ in range(num_nodes)]
    for node1 in range(num_nodes):
        for node2 in range(num_nodes):
            if node1!= node2:
                heuristic_matrix[node1][node2] = tentative_distances[node2][0]

    return heuristic_matrix
