def dijkstra(graph, start):
    """
    Dijkstra's algorithm for single-source shortest paths.
    """
    n = len(graph)
    dist = [float('inf')] * n
    dist[start] = 0
    prev = [-1] * n

    for _ in range(n):
        u = min(range(n), key=lambda x: dist[x])
        if dist[u] == float('inf'):
            break
        for v in range(n):
            if graph[u][v] and dist[u] + graph[u][v] < dist[v]:
                dist[v] = dist[u] + graph[u][v]
                prev[v] = u

    return dist
