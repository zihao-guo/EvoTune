def heuristics(distance_matrix, edge_weights):
    # Check if the dimensions of distance_matrix and edge_weights match
    if len(distance_matrix)!= len(edge_weights) or len(distance_matrix[0])!= len(edge_weights[0]):
        raise ValueError("Distance matrix and edge weights must have the same dimensions.")
    
    # Initialize heuristics matrix with the same shape as distance_matrix
    heuristics_matrix = distance_matrix.copy()
    
    # Iterate through each edge (i, j)
    for i in range(len(distance_matrix)):
        for j in range(len(distance_matrix[0])):
            # If there's an edge weight, incorporate it into the heuristic
            if edge_weights[i][j] is not None:
                # Calculate the cost to connect nodes i and j by adding the distance and the weight
                heuristics_matrix[i][j] += distance_matrix[i][j] + edge_weights[i][j]
            # If there's no edge weight, just add the distance
            else:
                heuristics_matrix[i][j] += distance_matrix[i][j]
    
    return heuristics_matrix
