```python
def heuristics(distance_matrix):
    num_nodes = distance_matrix.shape[0]
    
    # Calculate average distance from each node to all others
    avg_distance = distance_matrix.mean(axis=1).A1
    
    # Initialize the heuristics matrix with same dimensions as the distance_matrix
    heuristics_matrix = distance_matrix.copy()
    
    # Fill the heuristics matrix with CESP values
    for i in range(num_nodes):
        heuristics_matrix[i] += avg_distance[i]
        for j in range(num_nodes):
            if i!= j:
                heuristics_matrix[i] += distance_matrix[i, j]
    
    return heuristics_matrix
```
