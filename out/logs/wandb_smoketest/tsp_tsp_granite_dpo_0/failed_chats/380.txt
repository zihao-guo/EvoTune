def heuristics(distance_matrix):
    n = len(distance_matrix)
    
    # Initialize potential score matrix with zeros
    potential_score = [[0] * n for _ in range(n)]
    
    # Calculate potential score for each edge
    for i in range(n):
        for j in range(n):
            if i!= j:
                potential = distance_matrix[i][j]  # current edge distance
                
                # Calculate potential contribution
                reduction_potential = distance_matrix[i][i] + distance_matrix[j][j] - distance_matrix[i][j]
                potential += reduction_potential

                potential = potential * damping_factor
                
                potential_score[i][j] = potential
    
    return potential_score
