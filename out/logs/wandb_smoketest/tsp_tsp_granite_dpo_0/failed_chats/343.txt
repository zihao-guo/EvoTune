```python
def heuristics(distance_matrix):
    num_nodes = len(distance_matrix)
    heuristic_matrix = np.zeros((num_nodes, num_nodes))
    
    # Compute cost factor (average distance to other nodes)
    avg_distances = np.mean(distance_matrix, axis=1)
    heuristic_matrix += avg_distances[:, np.newaxis] + avg_distances[np.newaxis, :] - 2 * distance_matrix  # Each node's average distance to other nodes
    
    # Compute disparity factor (difference between a node's distance to neighbors and total distance)
    neighbors = np.argwhere(distance_matrix!= 0)
    heuristic_matrix += np.abs(distance_matrix[neighbors[:, 0], neighbors[:, 1]] - np.sum(distance_matrix, axis=1))
    
    return heuristic_matrix
```
