import numpy as np

def heuristics(distance_matrix):
    # Compute the square root of the distance matrix to work with distances
    distances = np.sqrt(distance_matrix)
    
    # Get the shape of the matrix
    rows, cols = distances.shape
    
    # Initialize the heuristic matrix with the same shape
    heuristic_matrix = np.zeros_like(distances)
    
    # Calculate the shortest path to every node from each node
    # Using Floyd-Warshall algorithm for efficiency
    for k in range(rows):
        for i in range(rows):
            for j in range(cols):
                heuristic_matrix[i][j] = min(heuristic_matrix[i][j], distances[i][k] + distances[k][j])
    
    return heuristic_matrix
