1. Direct distance (same row, same column): this could be the base cost of the edge.

3. Edge frequency: edges that appear frequently across the network might be less desirable.

```python
import numpy as np

def heuristics(distance_matrix, weights=[0.4, 0.3, 0.2, 0.1]):
    """
    Calculate a heuristic score for each edge in the distance matrix.

    Parameters:
    distance_matrix (numpy.ndarray): A square matrix of shape (n, n) where n is the number of nodes.
                                   The matrix should be symmetric and contain non-negative distances.
    weights (list, optional): A list of 4 weights for the different aspects of edge undesirability. 
                             Defaults to [0.4, 0.3, 0.2, 0.1].

    Returns:
    numpy.ndarray: A matrix of the same shape as the input, representing the heuristic score for each edge.
    """
    
    # Check if input is a valid square matrix
    if distance_matrix.shape!= (distance_matrix.shape[0], distance_matrix.shape[0]):
        raise ValueError("Input matrix should be a square matrix.")
    
    n = distance_matrix.shape[0]
    
    # Compute scores
    direct_cost = distance_matrix
    indirect_cost = np.minimum(distance_matrix[:, :-1], distance_matrix[:, 1:])  # Neighboring elements
    edge_frequency = np.full(distance_matrix.shape, 1/n)  # Assuming all edges appear once
    edge_length = np.full(distance_matrix.shape, np.sqrt(distance_matrix))  # Euclidean length of edges

    # Combine scores
    heuristic_scores = (
        weights[0] * direct_cost +
        weights[1] * np.minimum(indirect_cost, distance_matrix) +
        weights[2] * edge_frequency +
        weights[3] * edge_length  # Euclidean distance is used for simplicity; other measures could be employed
    )

    return heuristic_scores
```
