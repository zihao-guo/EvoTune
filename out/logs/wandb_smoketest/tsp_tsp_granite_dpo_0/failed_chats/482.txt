```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

def heuristics(distance_matrix):
    # First, compute the weighted degree centrality for each node
    degree_centrality = dict(zip(range(len(distance_matrix)),
                                     np.sum(distance_matrix, axis=0)))
    
    # For each edge (i, j), calculate cosine similarity between the degree vectors of (i)'s and (j)'s neighbors
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    for i in range(len(distance_matrix)):
        neighbors_i_degrees = [degree_centrality[j] for j in range(len(distance_matrix)) if distance_matrix[i, j]!= 0]
        neighbors_j_degrees = [degree_centrality[j] for j in range(len(distance_matrix)) if distance_matrix[j, i]!= 0]
        
        # Normalize the vectors and calculate cosine similarity
        norm_neighbors_i = np.array([np.sqrt(d) for d in neighbors_i_degrees])
        norm_neighbors_j = np.array([np.sqrt(d) for d in neighbors_j_degrees])
        similarity_ij = cosine_similarity([norm_neighbors_i, norm_neighbors_j])[0][1]
        
        heuristics_matrix[i, j] = similarity_ij
    
    return heuristics_matrix
```

This new heuristics function assigns a score to each edge based on the cosine similarity between the weighted degree vectors of the endpoints of the edge. The idea is to prioritize edges between nodes with similar degrees of importance, as they might carry crucial information.
