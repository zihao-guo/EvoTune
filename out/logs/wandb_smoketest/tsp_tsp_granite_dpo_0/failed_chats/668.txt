def heuristics(distance_matrix):
    import numpy as np

    # Get the dimensions of the input matrix
    n = distance_matrix.shape[0]

    # Compute the distance from each node to every other node
    # Using Floyd-Warshall algorithm for all-pairs shortest paths
    for k in range(n):
        for i in range(n):
            for j in range(n):
                distance_matrix[i][j] = min(distance_matrix[i][j], distance_matrix[i][k] + distance_matrix[k][j])

    # Create a parent array to trace the path from each node to the root
    parent = np.zeros((n, n), dtype=int)
    rank = [0] * n

    def find(x):
        if parent[x][x]!= x:
            parent[x] = find(parent[x][x])  # Path compression
        return parent[x][x]

    def union(x, y):
        root_x = find(x)
        root_y = find(y)

        if root_x == root_y:
            return

        if rank[root_x] > rank[root_y]:
            parent[root_y] = root_x
        elif rank[root_x] < rank[root_y]:
            parent[root_x] = root_y
        else:
            parent[root_y] = root_x
            rank[root_x] += 1

    # Kruskal's Algorithm using Key Path
    mst_edges = []
    i, e = 0, 0
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i][j] < float('inf'):
                e += 1
                mst_edges.append((i, j, distance_matrix[i][j]))
                union(i, j)

    return np.array(mst_edges).reshape(e, 3)
