```python
def cost_to_connect_heuristics(distance_matrix):
    # Get the dimensions of the matrix
    n = len(distance_matrix)
    
    # Calculate the cost-to-connect matrix
    cost_to_connect = np.full_like(distance_matrix, np.inf)
    
    # Fill the diagonal with 0 (connecting a node to itself has a cost of 0)
    np.fill_diagonal(cost_to_connect, 0)
    
    # Fill the upper triangular part with 1/distance
    cost_to_connect[np.triu_indices(n)] = 1 / distance_matrix[np.triu_indices(n)]
    
    # Fill the lower triangular part with the same values (since cost of connecting is symmetric)
    cost_to_connect[np.tril_indices(n, -1)] = cost_to_connect[np.tril_indices(n, -1)]
    
    return cost_to_connect
```
