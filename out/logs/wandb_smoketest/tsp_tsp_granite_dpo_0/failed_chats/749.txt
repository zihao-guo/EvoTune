```python
import numpy as np

def heuristics(distance_matrix):
    # Number of cities
    n = distance_matrix.shape[0]

    # Initialize the heuristic matrix with the inverse distance as the initial heuristic
    heuristic_matrix = -1 * np.diag(1 / distance_matrix)

    # Additional nearest neighbor heuristic
    for i in range(n):
        for j in range(n):
            if i!= j:
                heuristic_matrix[i, j] = min(distance_matrix[i, j], heuristic_matrix[i, :].min(), heuristic_matrix[:, j].min())

    return heuristic_matrix
```
