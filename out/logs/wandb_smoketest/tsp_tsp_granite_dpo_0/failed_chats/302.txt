```python
import numpy as np

def weighted_connectivity_density(distance_matrix):
    m, n = distance_matrix.shape
    # Calculate the number of connections for each node
    num_connections = np.sum(distance_matrix!= 0, axis=1)
    
    # Calculate the density (average distance between connected nodes)
    avg_dist_connected = np.zeros(m)
    for i in range(m):
        # Exclude the node from its own connections for calculating average distance
        mask = (distance_matrix[i]!= 0) & (distance_matrix[:, i]!= 0)
        avg_dist_connected[i] = np.mean(distance_matrix[mask]) if np.sum(mask) > 0 else 0

    # Compute the weighted heuristic for each edge
    heuristic_matrix = num_connections[:, None] - avg_dist_connected[None, :]

    # Take the absolute value and flip the sign for lower is better
    heuristic_matrix = np.abs(heuristic_matrix).T  # Transpose for consistency with distance matrix shape
    heuristic_matrix *= -1  # Flip the sign

    return heuristic_matrix
```
