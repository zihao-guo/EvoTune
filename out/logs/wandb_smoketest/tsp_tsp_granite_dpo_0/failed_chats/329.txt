import numpy as np

def heuristics(distance_matrix):
    # Compute the adjacency matrix from the distance matrix
    n = distance_matrix.shape[0]
    adj_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            adj_matrix[i, j] = adj_matrix[j, i] = distance_matrix[i, j]
            
    # Calculate the degree of each node
    degrees = np.sum(adj_matrix, axis=1)

    # Compute edge density as a heuristic score
    score = 1 - np.minimum(degrees, 1) / n

    return score
