```python
import numpy as np

def relaxation_heuristics(distance_matrix):
    # Calculate the inverse of each element in the distance matrix
    inv_distances = 1 / distance_matrix
    
    # Apply softmax to get the probabilities
    sum_inv_distances = np.sum(inv_distances, axis=0)
    exp_inv_distances = np.exp(inv_distances - sum_inv_distances[:, np.newaxis])
    probs = exp_inv_distances / np.sum(exp_inv_distances, axis=0)
    
    # The resulting matrix now represents the undesirability of including each edge
    return probs
```
