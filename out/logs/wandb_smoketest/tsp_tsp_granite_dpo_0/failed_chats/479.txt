import heapq
import numpy as np

def heuristics(distance_matrix):
    # Step 1: Floyd-Warshall algorithm to compute shortest paths between all pairs of nodes
    n = distance_matrix.shape[0]
    shortest_paths = distance_matrix.copy()
    for k in range(n):
        for i in range(n):
            for j in range(n):
                shortest_paths[i, j] = min(shortest_paths[i, j], shortest_paths[i, k] + shortest_paths[k, j])
    
    # Step 2: Create a priority queue (min-heap) to store edges, where priority is determined by the potential cost to reach the last node
    # Each edge (i, j) is represented as a tuple (i, j, current_cost + shortest_path from i to j)
    edge_costs = []
    for i in range(n):
        for j in range(i+1, n):  # Only consider edges between different nodes
            current_cost = shortest_paths[i, j]
            shortest_path_from_i = shortest_paths[i, n-1]  # Assume the last node is 'n-1'
            total_cost = current_cost + shortest_path_from_i
            edge_costs.append((i, j, total_cost))
    
    # Step 3: Use heapq to sort the edges by their total cost
    heapq.heapify(edge_costs)

    # Step 4: Create a matrix to store the undesirability of each edge
    undesirability = np.zeros_like(distance_matrix)
    
    # Pop elements from the priority queue and assign their costs to the corresponding edges in the undesirability matrix
    for _ in range(n):
        _, _, current_cost = heapq.heappop(edge_costs)
        for i in range(n):
            for j in range(i+1, n):
                undesirability[i, j] = current_cost
                undesirability[j, i] = current_cost

    return undesirability
