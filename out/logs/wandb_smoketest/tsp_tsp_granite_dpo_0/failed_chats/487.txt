```python
def heuristics(distance_matrix):
    n = len(distance_matrix)
    # Calculate topological sort and indegree
    indegree = [0]*n
    for i in range(n):
        for j in range(i):
            if distance_matrix[i][j]!= 0:
                indegree[j] += 1

    # Create a list to hold edges with negative weight
    edges_with_negatives = []
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i][j] < 0 and indegree[i] > 0 and indegree[j] == 0:
                edges_with_negatives.append((i, j))

    # Calculate heuristics score for each edge
    heuristics_matrix = [[0]*n for _ in range(n)]
    for i, j in edges_with_negatives:
        heuristics_matrix[i][j] = distance_matrix[i][j]
        heuristics_matrix[j][i] = distance_matrix[j][i]

    # Add more weight to edges with lower indegree (higher importance)
    for i in range(n):
        heuristics_matrix[i][i] = indegree[i]

    # Normalize the matrix
    max_score = max(max(row) for row in heuristics_matrix)
    return [max_score if heuristics_matrix[i][j] == 0 else heuristics_matrix[i][j]/max_score for i in range(n) for j in range(n)]
```
