import numpy as np

def heuristics(distance_matrix):

    heuristic_matrix = np.zeros((n, n))

    for i in range(n):
        for j in range(n):
            if i!= j:  # Don't consider the diagonal elements
                path_cost = np.sum(distance_matrix[i]) + np.sum(distance_matrix[j]) - 2 * distance_matrix[i][j]
                heuristic_matrix[i][j] = max(path_cost - 1, 0)  # Adjust for a non-zero score

    # Adjust higher-diagonal elements to discourage inclusion of more distant edges
    for k in range(1, n):
        for i in range(n):
            for j in range(i+1, n):
                heuristic_matrix[i][j] = min(heuristic_matrix[i][j], heuristic_matrix[i][k] + heuristic_matrix[k][j] - 1)

    return heuristic_matrix
