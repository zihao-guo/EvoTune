import numpy as np

def weighted_heuristics(distance_matrix, weights=None):
    """
    This function calculates a new heuristic matrix based on the input distance_matrix,
    applying a weighted distance approach and a penalty for high-weight edges.
    The weights can be provided as a separate array of the same shape as the distance_matrix.
    """
    if weights is None:
        # If weights are not provided, assign equal weights to all edges
        weights = np.ones_like(distance_matrix)
    
    # Apply a penalty for high-weight edges
    penalty = 10  # Adjust this value according to the desired penalty strength
    high_weight_penalty = np.where(weights > 0.5, penalty, 0)
    weighted_distance = distance_matrix * weights
    heuristic_matrix = weighted_distance + high_weight_penalty
    
    return heuristic_matrix
