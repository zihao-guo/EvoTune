def heuristics(distance_matrix):

    n = distance_matrix.shape[0]
    shortest_paths = distance_matrix.copy()
    for k in range(n):
        for i in range(n):
            for j in range(n):
                shortest_paths[i, j] = min(shortest_paths[i, j], shortest_paths[i, k] + shortest_paths[k, j])

    # Applying the Christofides-Serdyukov algorithm to get an approximate TSP tour and its edge weights
    # The Christofides-Serdyukov algorithm finds a 1.5-approximation for the TSP problem
    # We'll ignore the tour and focus on the edge weights as our heuristic values
    n_approx = int(1.5 * n)
    unvisited = set(range(n))
    tour = []
    edge_weights = []

    while len(tour) < n_approx:
        u = min(unvisited, key=lambda x: shortest_paths[x, -1])
        tour.append(u)
        unvisited.remove(u)
        for v in unvisited:
            if shortest_paths[u, v] + shortest_paths[-1, v] == shortest_paths[u, -1]:
                edge_weights.append(shortest_paths[u, v])
                break

    heuristic_values = np.array(edge_weights)

    return heuristic_values
