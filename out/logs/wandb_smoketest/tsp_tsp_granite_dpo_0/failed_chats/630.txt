import numpy as np

def heuristics(distance_matrix):

def _betweenness_centrality(distance_matrix):
    n = len(distance_matrix)
    betweenness = np.zeros((n, n))
    
    for i in range(n):
        # Compute the number of shortest paths from source to all other nodes
        num_paths = np.sum(np.linalg.inv(distance_matrix + np.eye(n)), axis=0)
        
        # Compute the total number of paths from source to all other nodes
        total_paths = np.sum(np.linalg.inv(distance_matrix + np.eye(n)), axis=1)
        
        # Compute betweenness score for each node
        betweenness[i] = np.sum(num_paths / total_paths)
    
    return betweenness

def _closeness_centrality(distance_matrix):
    n = len(distance_matrix)
    closeness = np.zeros(n)
    
    for i in range(n):
        # Compute the average shortest path distance to all other nodes
        avg_path_length = np.sum(np.linalg.inv(distance_matrix + np.eye(n))[i, :]) / (n - 1)
        
        # Compute the closeness score
        closeness[i] = n / avg_path_length
    
    return closeness
