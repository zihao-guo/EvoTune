from scipy.sparse.csgraph import shortest_path

    num_edges = distance_matrix.size // 2

    dist_matrix, _ = shortest_path(distance_matrix, directed=False)

    heuristic_matrix = np.zeros_like(distance_matrix)
    
    for i in range(distance_matrix.shape[0]):
        for j in range(i, distance_matrix.shape[1]):
            # Exclude diagonal elements (self-loops)
            if i!= j:
                heuristic_matrix[i, j] = degrees[i] * (num_edges - degrees[i]) * dist_matrix[i, j]
                heuristic_matrix[j, i] = heuristic_matrix[i, j]
    
    return heuristic_matrix
