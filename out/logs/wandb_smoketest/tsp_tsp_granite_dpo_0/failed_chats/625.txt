def advanced_heuristics(distance_matrix):

    # Initialize the MST and priority queue
    mst = [(-1, -1) for _ in range(n)]  # (-weight, vertex)
    parent = [None] * n
    min_heap = [(0, 0)]  # (weight, current_vertex)
    
    # Initialize the visited set
    visited = set()
    
    # Priority queue operations
    while min_heap:
        weight, current_vertex = heapq.heappop(min_heap)
        
        # If vertex is already visited, skip it
        if current_vertex in visited:
            continue
        
        # Mark current vertex as visited
        visited.add(current_vertex)
        
        # Traverse the graph to find the minimum-weight edge to this vertex
        for next_vertex, edge_weight in enumerate(distance_matrix[current_vertex]):
            if edge_weight!= float('inf') and next_vertex not in visited:

                    mst[next_vertex] = (weight + edge_weight, next_vertex)
                    parent[next_vertex] = current_vertex
                    heapq.heappush(min_heap, (weight + edge_weight, next_vertex))
    
    # Construct the MST from the minimum-weight edges
    mst_edges = [(mst[i][1], mst[i][0], i) for i in range(n) if parent[i] is not None]
    return mst_edges
