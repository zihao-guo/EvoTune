```python
import numpy as np

def heuristics(distance_matrix):
    # Calculate the sum of the distance matrix
    total_distance = np.sum(distance_matrix)
    
    # Calculate the number of connections
    num_connections = np.count_nonzero(distance_matrix)
    
    # Calculate the Weighted Connection Score (WCS)
    # WCS is the total distance divided by the number of connections
    # This will penalize frequently connected edges more
    wcs_matrix = total_distance / num_connections
    
    return wcs_matrix
```
