def heuristics(distance_matrix):

    num_vertices = len(distance_matrix)

    heuristic_matrix = [[0] * num_vertices for _ in range(num_vertices)]

    for i in range(num_vertices):
        for j in range(i+1, num_vertices):
            heuristic_matrix[i][j] = distance_matrix[i][j]  # same as input
    
    # Calculate Max-Heuristic (longest edges)
    for i in range(num_vertices):
        for j in range(i+1, num_vertices):
            max_distance = float('-inf')
            for k in range(num_vertices):
                if i!= k and j!= k:
                    max_distance = max(max_distance, distance_matrix[i][k] + distance_matrix[k][j])
            heuristic_matrix[i][j] = max_distance  # update with longest edges
    
    # Create a new heuristic matrix that balances both heuristics
    # Using a weighted sum of Min-Heuristic and Max-Heuristic

    weighted_heuristic_matrix = [[0] * num_vertices for _ in range(num_vertices)]
