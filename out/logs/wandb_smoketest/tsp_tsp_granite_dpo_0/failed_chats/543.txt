def heuristics(distance_matrix, edge_importance):

    Returns:
    A cost matrix where each entry represents the combined cost of including an edge,
    considering both its distance and the importance of the edge.
    """
    # Multiply the distance matrix by the edge importance matrix
    weighted_distance_matrix = [
        [distance * importance for distance in row] for row in distance_matrix
    ]

    # Sum up each row to get the total cost for each edge
    cost_matrix = [sum(row) for row in weighted_distance_matrix]

    return cost_matrix
