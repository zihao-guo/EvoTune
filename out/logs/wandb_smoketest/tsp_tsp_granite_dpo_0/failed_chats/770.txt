def heuristics(distance_matrix):
    import numpy as np

    # Calculate the shortest path from each node to all others using Dijkstra's algorithm
    def dijkstra(matrix):
        n = matrix.shape[0]
        distances = np.copy(matrix)
        visited = np.zeros(n, dtype=bool)
        for _ in range(n):
            min_distance = np.inf
            u = -1
            for v in range(n):
                if not visited[v] and distances[v] < min_distance:
                    min_distance = distances[v]
                    u = v
            visited[u] = True
            for v in range(n):
                if visited[v] and distances[v] > distances[u] + matrix[u, v]:
                    distances[v] = distances[u] + matrix[u, v]
        return distances

    # Calculate the distance impact factor (DIF) for each edge
    DIF = np.zeros(distance_matrix.shape)
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            DIF[i, j] = distance_matrix[i, j] / np.sum(dijkstra(distance_matrix))

    return DIF
