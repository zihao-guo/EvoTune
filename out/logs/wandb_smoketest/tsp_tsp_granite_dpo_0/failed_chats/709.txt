def heuristics(distance_matrix):
    import numpy as np
    
    # Step 1: Calculate inverse distances (higher inverse = shorter paths)
    inverse_distances = 1.0 / distance_matrix
    
    # Step 2: Normalize the inverse distances to sum up to 1

    return heuristic_scores
