def heuristics(distance_matrix):

    # Combine deviation score and penalty score
    heuristic_score = deviation_score + weight_penalty_score

    # Normalize the score to a range between 0 and 1
    max_score = n * n  # Maximum possible deviation score (n^2)

    return normalized_score
