```python
import numpy as np

def triangle_inequality_heuristics(distance_matrix):
    n = distance_matrix.shape[0]  # number of nodes
    heuristic_matrix = np.zeros_like(distance_matrix)

    for i in range(n):
        for j in range(n):
            # Get all neighboring nodes k
            k_indices = np.where(distance_matrix[i, :] < np.inf)[0]
            
            # Only consider valid neighbors (not including i or j)
            k_indices = k_indices[(k_indices!= i) & (k_indices!= j)]
            
            if len(k_indices) > 0:
                # Find the minimum distance through any k
                min_through_k = min([distance_matrix[k, j] for k in k_indices])
                
                # Update the heuristic for the edge (i, j)
                heuristic_matrix[i, j] = min(distance_matrix[i, j], distance_matrix[i, k] + distance_matrix[k, j] - min_through_k)
    
    return heuristic_matrix
```
