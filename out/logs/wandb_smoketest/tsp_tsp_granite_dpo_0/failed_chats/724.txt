import networkx as nx

def heuristics(distance_matrix):
    # Convert distance_matrix to adjacency matrix
    G = nx.from_numpy_array(distance_matrix)
    
    # Calculate shortest path lengths
    shortest_paths = nx.all_pairs_shortest_path_length(G)
    
    # Calculate edge betweenness
    edge_betweenness = nx.edge_betweenness_centrality(G, weight='weight')  # Assuming 'weight' is the original distance
    
    # Adaptive decay function
    # Simplified version: exponentially decreases with distance
    decay = 2 ** (G.number_of_nodes() - 1 - G.number_of_edges() / 2)
    
    # Combine all indicators
    heuristics_matrix = (
        (shortest_paths[u][v] + 1) * edge_betweenness[u, v] * decay
        for u, v in G.edges()
    )
    
    # Normalize heuristics values to a reasonable range
    return [min(x) for x in heuristics_matrix]
