The idea is to first cluster the vertices (nodes) in the graph based on their proximity (represented by the distance matrix), then calculate the edge-betweenness centrality for edges within each cluster to estimate their importance.

Here's the Python function that implements this approach:

```python
import numpy as np
from scipy.cluster.hierarchy import linkage, dendrogram
from scipy.spatial.distance import pdist

def edge_betweenness_centrality(distance_matrix):
    n = distance_matrix.shape[0]
    adj_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if distance_matrix[i, j]!= 0:
                adj_matrix[i, j] = distance_matrix[i, j]
                adj_matrix[j, i] = distance_matrix[i, j]  # Undirected graph

    # Calculate edge-betweenness centrality using the clustering hierarchy
    z = linkage(pdist(adj_matrix), method='complete')
    f = 1 / (4 * np.pi * np.linalg.norm(z[:, 2:])**2)
    bc = np.sum(f * z[:, 2:], axis=1)

    return bc

def heuristics(distance_matrix):
    bc = edge_betweenness_centrality(distance_matrix)
    return bc
```
