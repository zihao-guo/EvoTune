```python
import heapq

def kruskal_heuristic(distance_matrix):
    # Initialize a graph and a priority queue to manage edges
    graph = {i: [] for i in range(len(distance_matrix))}
    for i in range(len(distance_matrix)):
        for j in range(i+1, len(distance_matrix)):
            graph[i].append((distance_matrix[i][j], i, j))
            graph[j].append((distance_matrix[i][j], i, j))
    
    # Initialize a set to keep track of edges in MST
    mst = set()
    
    # Initialize a list to keep track of edges by weight
    edge_weight = []

    # Kruskal's Algorithm with Weighted Edge Addition Factor
    for edge in sorted(graph[0], key=lambda x: x[0]):  # Start with the smallest edge
        if not any(edge[1] in mst or edge[2] in mst, graph[edge[1]], graph[edge[2]]):
            mst.add(edge)
            heapq.heappush(edge_weight, edge[0])
            
            # Add edges with lower weights to MST
            while edge_weight and len(mst) > 1 and edge_weight[0] < min(edge_weight[0], next((w for e in graph[edge[1]] if e[2] not in mst), (float('inf'), 0, 0)))[0]:
                mst.add(next((e for e in graph[edge[1]] if e[2] not in mst), (float('inf'), 0, 0)))
                heapq.heappush(edge_weight, next((w for e in graph[edge[1]] if e[2] not in mst), (float('inf'), 0, 0))[0])
                mst.remove((edge[1], edge[2]))

    return [min(edge[0] for edge in mst) for edge in mst]
```

This function not only maintains the graph structure like the original one, but also integrates a weighted edge addition factor. It first picks the smallest edge, then adds edges with the next smallest weights (but only if they don't form a cycle with edges already in the MST).
