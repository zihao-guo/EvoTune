import heapq

def heuristics(distance_matrix):
    n = len(distance_matrix)
    # Initialize the result matrix with the same shape as the input
    heuristic_matrix = [[0] * n for _ in range(n)]

    # Create priority queue and distance dictionary
    priority_queue = [(0, 0)]  # (current distance, current node)
    shortest_paths = {i: float('inf') for i in range(n)}
    shortest_paths[0] = 0

    while priority_queue:
        current_distance, current_node = heapq.heappop(priority_queue)

        # If we've found a shorter path to the current node, update
        if current_distance > shortest_paths[current_node]:
            continue

        # Explore neighbors
        for neighbor, weight in enumerate(distance_matrix[current_node]):
            if weight + shortest_paths[current_node] < shortest_paths[neighbor]:
                shortest_paths[neighbor] = weight + shortest_paths[current_node]
                heapq.heappush(priority_queue, (shortest_paths[neighbor], neighbor))

        # Assign heuristic values based on shortest path distances
        for i in range(n):
            for j in range(n):
                if i == j:
                    heuristic_matrix[i][j] = distance_matrix[i][j]
                else:
                    heuristic_matrix[i][j] = shortest_paths[i] + distance_matrix[i][j]

    return heuristic_matrix
