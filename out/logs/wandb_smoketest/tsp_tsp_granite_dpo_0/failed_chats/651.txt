import numpy as np
from sklearn.cluster import AgglomerativeClustering
from sklearn.preprocessing import normalize

def heuristics(distance_matrix):
    # Normalize the distance matrix
    norm_dist = normalize(distance_matrix, norm='l2', axis=0)
    
    # Apply hierarchical clustering (AgglomerativeClustering with linkage 'ward')
    clustering = AgglomerativeClustering(n_clusters=None, linkage='ward')
    clusters = clustering.fit_predict(norm_dist)
    
    # Create a similar-shaped output matrix to store heuristic scores
    heuristic_score_matrix = np.zeros(distance_matrix.shape)
    
    # Assign heuristic scores based on density and connectivity within clusters
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            if clusters[i] == clusters[j]:
                # If both vertices are in the same cluster, they are likely 'desirable'
                heuristic_score_matrix[i, j] = -np.log(distance_matrix[i, j] / np.mean(distance_matrix[clusters == clusters[i]]))
            else:
                # If they are in different clusters, they are likely 'undesirable'
                heuristic_score_matrix[i, j] = distance_matrix[i, j]
    
    return heuristic_score_matrix
