import numpy as np
from sklearn.cluster import SpectralClustering

def heuristics(distance_matrix):

    # Use Spectral Clustering on the similarity matrix
    clustering = SpectralClustering(n_clusters=len(distance_matrix), affinity="precomputed", assign_labels="hierarchical").fit(similarity_matrix)

    # Assign a higher undesirability score to edges between clusters
    undesirability_score_matrix = np.zeros_like(distance_matrix)
    for i in range(len(distance_matrix)):
        for j in range(i, len(distance_matrix)):
            if clustering.labels_[i]!= clustering.labels_[j]:
                undesirability_score_matrix[i, j] = 1.0  # Set score to 1 for edges between clusters
            else:

                undesirability_score_matrix[i, j] = 1.0 - similarity_matrix[i, j]  # Higher values indicate more undesirability

    return undesirability_score_matrix
