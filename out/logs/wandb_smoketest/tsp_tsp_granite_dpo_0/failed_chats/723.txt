def heuristics(distance_matrix):
    n = len(distance_matrix)
    heuristic_matrix = [[0.0] * n for _ in range(n)]

    # Calculate the heuristic value for each edge
    for i in range(n):
        for j in range(i, n):
            # Find the minimum cost to reach nodes i and j from the rest of the graph
            min_cost_path_i = [float('inf')] * n
            min_cost_path_j = [float('inf')] * n
            for k in range(n):

                    min_cost_path_i[k] = min(min_cost_path_i[k], distance_matrix[i][k] + min_cost_path_i[k])
                    min_cost_path_j[k] = min(min_cost_path_j[k], distance_matrix[j][k] + min_cost_path_j[k])
            
            # Calculate the potential saving by inserting the edge (i, j)
            saving = min(min_cost_path_i[j], min_cost_path_j[i]) - distance_matrix[i][j]

            heuristic_matrix[i][j] = abs(saving)
    
    return heuristic_matrix
