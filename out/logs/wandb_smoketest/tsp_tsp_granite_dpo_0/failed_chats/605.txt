import numpy as np

def heuristics(distance_matrix):
    n = len(distance_matrix)
    heuristics_matrix = np.copy(distance_matrix)  # Keep the original matrix

    # Select a random edge
    i, j = np.random.choice(n, size=2, replace=False)
    min_distance = min(distance_matrix[i, :], distance_matrix[:, j])

    # Find the optimal replacement for the selected edge
    current_node = i
    for k in range(n):
        if distance_matrix[current_node, k] < min_distance:
            current_node = k

    current_path = [i]
    new_path = [current_node]

    while current_node!= j:
        min_distance_adj = min(distance_matrix[current_node, new_path[-1]], 
                               distance_matrix[new_path[-1], current_node])
        if min_distance_adj < distance_matrix[current_node, new_path[-1]]:
            new_path.append(current_node)
            current_node = min_distance_adj
        else:
            new_path.append(new_path[-1])
            current_node = new_path[-1]

    new_path.append(j)

    # Calculate the heuristic score
    new_distance = sum(distance_matrix[p, q] for p, q in zip(new_path, new_path[1:]))
    heuristics_matrix[i, j] = new_distance

    return heuristics_matrix
