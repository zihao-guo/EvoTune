```python
import numpy as np

def betweenness_centrality_heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    adjacency_matrix = np.zeros((n, n))

    # Construct adjacency matrix
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j]!= 0:  # only consider non-zero distances
                adjacency_matrix[i, j] = 1
                adjacency_matrix[j, i] = 1

    # Calculate betweenness centrality
    betw_centrality = np.zeros((n, n))
    for _ in range(2 * n - 2):  # 2*n-2 is the number of edges in a complete graph
        for j in range(n):
            delta = np.copy(adjacency_matrix)
            delta[j] = 0
            delta[j][j] = 0
            delta[j] += np.eye(n) * adjacency_matrix[j]
            betw_centrality += (adjacency_matrix * (np.linalg.matrix_power(delta, 2) - delta)) * (1 / np.sum(adjacency_matrix))

    # Normalize betweenness centrality to get undesirability score
    betw_centrality_norm = np.maximum(betw_centrality, 0) / np.sum(betw_centrality)

    # Ensure result is of the same shape as input distance matrix
    betw_centrality_norm = betw_centrality_norm.reshape(distance_matrix.shape)

    return betw_centrality_norm
```

Finally, it normalizes these values to ensure they sum to 1 (which can be viewed as a probability of edge undesirability), and ensures the output is of the same shape as the input distance matrix.
