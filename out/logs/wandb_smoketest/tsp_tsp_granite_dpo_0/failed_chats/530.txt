import numpy as np

def structural_heuristics(distance_matrix):
    # Number of nodes in the graph
    num_nodes = distance_matrix.shape[0]

    # Calculate graph density
    density = np.sum(distance_matrix) / (num_nodes * (num_nodes - 1))
    
    # Calculate connectedness (number of connected components)
    # We use the reverse of the distance matrix and find the connected components using Depth-First Search (DFS)
    reverse_dist_matrix = np.min(distance_matrix, axis=0)
    connected_components = len(np.unique(np.concatenate(np.where(reverse_dist_matrix!= np.inf, 
                                                np.array([np.searchsorted(reverse_dist_matrix, 
                                                                               np.inf) + 1, 
                                                                -1]), 
                                                axis=1))))

    # Create an edge undesirability score matrix
    heuristics_matrix = np.zeros(distance_matrix.shape)

    # Populate heuristics matrix based on edge weights and connectedness
    for i in range(num_nodes):
        for j in range(i+1, num_nodes):
            # If nodes i and j are connected
            if connected_components > 1:
                # Calculate the weighted sum of neighbors' distances
                weight_sum = np.sum(distance_matrix[i, :]) + np.sum(distance_matrix[j, :])
                # If this edge is heavier than its neighbors, increase its undesirability score
                if distance_matrix[i, j] > weight_sum:
                    heuristics_matrix[i, j] = (1 - density) * (1 - weight_sum / np.sum(distance_matrix[i, :]))
                    heuristics_matrix[j, i] = heuristics_matrix[i, j]  # Assuming undirected graph

    return heuristics_matrix
