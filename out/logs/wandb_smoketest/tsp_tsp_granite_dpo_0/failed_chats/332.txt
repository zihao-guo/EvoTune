import heapq

def heuristics(distance_matrix):
    # Initializing the graph with an infinite cost for all nodes
    graph = {node: {node: float('inf') for node in graph} for node in range(len(distance_matrix))}
    for i in range(len(distance_matrix)):
        for j in range(len(distance_matrix)):
            if i!= j:
                graph[i][j] = distance_matrix[i][j]
    
    # Setting the cost of a node to itself to be zero
    for i in range(len(graph)):
        graph[i][i] = 0
    
    # Dijkstra's algorithm to find the shortest paths
    paths = {node: float('inf') for node in graph}
    paths[0] = 0
    priority_queue = [(0, 0)]  # (current_cost, current_node)
    
    while priority_queue:
        current_cost, current_node = heapq.heappop(priority_queue)
        
        # Skip if we've found a better path to the current node
        if current_cost > paths[current_node]:
            continue
        
        # Update the shortest path costs for neighboring nodes
        for neighbor, weight in graph[current_node].items():
            new_cost = current_cost + weight
            if new_cost < paths[neighbor]:
                paths[neighbor] = new_cost
                heapq.heappush(priority_queue, (new_cost, neighbor))
    
    # Create a heuristics matrix
    heuristics_matrix = [[paths[i] + graph[i][j] for j in range(len(distance_matrix))] for i in range(len(distance_matrix))]
    
    return heuristics_matrix
