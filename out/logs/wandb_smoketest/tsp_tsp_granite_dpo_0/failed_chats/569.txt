def heuristics(distance_matrix):
    import numpy as np

    # Calculate the triangle inequality cost
    n = len(distance_matrix)
    triangle_inequality_costs = np.zeros_like(distance_matrix)

    for i in range(n):
        for j in range(n):
            for k in range(n):
                if i!= j and j!= k and i!= k:
                    triangle_inequality_costs[i, j] = min(
                        triangle_inequality_costs[i, j],
                        distance_matrix[i, k] + distance_matrix[k, j]
                    )

    # Calculate the heuristic values
    heuristic_values = np.zeros_like(distance_matrix)

    for i in range(n):
        for j in range(n):
            heuristic_values[i, j] = distance_matrix[i, j] + triangle_inequality_costs[i, j] - min(
                distance_matrix[i, j], triangle_inequality_costs[i, j]
            )

    # Normalize heuristic values to ensure they range between 0 and 1
    heuristic_values = (heuristic_values - min(heuristic_values.min(), triangle_inequality_costs.min())) / (
        max(heuristic_values.max(), triangle_inequality_costs.max()) - min(heuristic_values.min(), triangle_inequality_costs.min())

    return heuristic_values
