    improved = distance_matrix.copy()

        for j in range(i+1, n):

            min_cost = float('inf')
            improved_edge = (None, None)

                if k not in (i, j):
                    cost = distance_matrix[i][j] + distance_matrix[k][j] + distance_matrix[i][k]
