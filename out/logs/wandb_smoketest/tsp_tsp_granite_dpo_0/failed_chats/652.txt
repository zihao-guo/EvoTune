```python
def heuristics(distance_matrix):
    num_nodes = distance_matrix.shape[0]
    
    # Degree of connectivity heuristic
    degree = np.sum(distance_matrix, axis=0)
    
    # Proximity to optimality heuristic
    optimal_distance = np.min(np.min(distance_matrix, axis=0))
    proximity = np.inf
    for i in range(num_nodes):
        for j in range(i+1, num_nodes):
            dist = distance_matrix[i, j]
            proximity = min(proximity, dist + optimal_distance)
    
    # Combine heuristics
    heuristics_matrix = degree + proximity - (distance_matrix.diagonal() * 2)
    
    return heuristics_matrix
```
