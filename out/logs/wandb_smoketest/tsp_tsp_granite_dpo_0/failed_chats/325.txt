The PSP is a heuristic that estimates the distance from a node to all other nodes based on the shortest path that can be reached from that node. In this context, edges with higher PSPs will be considered more undesirable because they potentially limit the flexibility to improve the overall path later.

The new heuristics function, `psp_heuristics()`, will do the following:

1. Convert the given distance matrix into a graph, where the weights represent the distances.
2. Compute the PSPs for each node.
3. Calculate a new heuristic score for each edge based on the PSPs and the original distance.

Here's the Python function:

```python
import numpy as np

def psp_heuristics(distance_matrix):
    # Step 1: Convert distance matrix to graph
    num_nodes = distance_matrix.shape[0]
    graph = np.zeros((num_nodes, num_nodes))
    for i in range(num_nodes):
        for j in range(num_nodes):
            if i!= j:
                graph[i][j] = distance_matrix[i][j]
    
    # Step 2: Compute PSPs
    def psp(node, graph):
        short_paths = []
        for other_node in range(num_nodes):
            if other_node!= node:
                path_length = float('inf')
                path = []
                while other_node!= node:
                    other_node = min((n for n in range(num_nodes) if n!= node and graph[node][n]!= float('inf')), key=lambda x: graph[node][x])
                    path.append(other_node)
                    path_length = min(path_length, graph[node][other_node])
                    node = other_node
                path.append(node)
                path.reverse()
                short_paths.append((path, path_length))
        return short_paths

    psp_values = psp(0, graph)
    
    # Step 3: Calculate new heuristic score for each edge
    heuristic_scores = np.zeros_like(distance_matrix)
    for i in range(num_nodes):
        for j in range(num_nodes):
            if i!= j:
                heuristic_scores[i][j] = distance_matrix[i][j] + min([psp_value[1] for psp_value in psp_values[i] if j in psp_value[0]])
    
    return heuristic_scores
```
