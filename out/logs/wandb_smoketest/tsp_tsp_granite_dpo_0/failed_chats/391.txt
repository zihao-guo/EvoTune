import numpy as np

def potential_energy(distance_matrix):
    # This function calculates the potential energy matrix.
    # It considers the negative of the sum of distances for each node.
    n = distance_matrix.shape[0]
    potential_energy = np.zeros(distance_matrix.shape)
    for i in range(n):
        for j in range(n):
            if i!= j:
                potential_energy[i][j] = potential_energy[j][i] = -distance_matrix[i][j]
    return potential_energy

def heuristics(distance_matrix):
    # This is the function that uses the potential energy as heuristic.
    potential_energies = potential_energy(distance_matrix)
    # Normalize potential energies to form a valid heuristic matrix.
    max_potential = np.max(potential_energies)
    normalized_heuristics = potential_energies / max_potential
    return normalized_heuristics
