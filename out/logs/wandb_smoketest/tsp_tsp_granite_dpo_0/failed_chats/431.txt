def heuristics(distance_matrix):
    import numpy as np
    import random

    # Step 1: Convert distance matrix into a list of edges
    num_vertices = len(distance_matrix)
    edges = [(i, j, distance_matrix[i][j]) for i in range(num_vertices) for j in range(i+1, num_vertices)]

    def swap_edges(edges, e1, e2):
        nonlocal new_edges
        new_edges = edges[:]
        new_edges.remove(e1)
        new_edges.remove(e2)
        new_edges.append((e1[0], e2[1], e1[2] + e2[2]))
        new_edges.append((e1[1], e2[0], e1[2] + e2[2]))

    def improve(edges, new_edges):
        cost = 0
        for e in edges:
            cost += e[2]
        for e in new_edges:
            cost -= e[2]
        return cost

    # Step 2: Implement basic Lin-Kernighan heuristic
    max_iter = 1000
    current_edges = edges.copy()
    best_cost = improve(current_edges, current_edges)

    for _ in range(max_iter):
        # Step 3: Perform edge swaps
        e1 = random.choice(current_edges)
        e2 = random.choice(current_edges)
        swap_edges(current_edges, e1, e2)
        new_cost = improve(current_edges, current_edges)

        if new_cost < best_cost:
            best_cost = new_cost
            current_edges = current_edges.copy()  # Copy to preserve original edges

    # Step 4: Return the undesirability scores for each edge
    return [(e[1], best_cost - e[2]) for e in current_edges]
