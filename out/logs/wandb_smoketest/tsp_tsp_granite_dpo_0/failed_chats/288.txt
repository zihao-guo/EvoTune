Negative cycles in a graph correspond to edges with negative total weight, which are essentially "bad" choices in an optimization problem like the Traveling Salesman Problem (TSP). import numpy as np

def floyd_warshall(distance_matrix):
    num_nodes = distance_matrix.shape[0]
    for k in range(num_nodes):
        for i in range(num_nodes):
            for j in range(num_nodes):
                distance_matrix[i][j] = min(distance_matrix[i][j], distance_matrix[i][k] + distance_matrix[k][j])
    return distance_matrix

def heuristics(distance_matrix):

    fw_dist_matrix = floyd_warshall(distance_matrix)
    
    # Add diagonal with negative infinity to prevent negative cycles
    num_nodes = distance_matrix.shape[0]
    fw_dist_matrix = np.vstack((fw_dist_matrix, np.full((1, num_nodes), -np.inf)))

    n_cycles = 0

    # Add the number of detected negative cycles as heuristics to the original distance matrix
