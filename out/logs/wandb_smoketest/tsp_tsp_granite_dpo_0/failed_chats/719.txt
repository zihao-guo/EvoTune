2. **Negative Cycle Detection**: If a negative cycle is detected in the distance matrix, edges within this cycle should be given extremely low heuristics scores, as they can make the solution suboptimal or unsolvable.

import heapq

def heuristics(distance_matrix):
    n = len(distance_matrix)
    if n == 0:
        return distance_matrix
    
    # Initialize shortest paths using Dijkstra's Algorithm
    shortest_paths = [float('inf')] * n
    shortest_paths[0] = 0
    priority_queue = [(0, 0)]  # (distance, node)
    
    while priority_queue:
        current_distance, current_node = heapq.heappop(priority_queue)
        
        if current_distance > shortest_paths[current_node]:
            continue
        
        for neighbor, weight in enumerate(distance_matrix[current_node]):
            if shortest_paths[neighbor] > current_distance + weight:
                shortest_paths[neighbor] = current_distance + weight
                heapq.heappush(priority_queue, (shortest_paths[neighbor], neighbor))
    
    # Calculate the heuristics
    heuristics_matrix = [min(shortest_paths[i], 1.0 / shortest_paths[i]) for i in range(n)]
    
    # Negative cycle detection
    for i in range(n):
        for j in range(n):
            for k in range(n):
                if distance_matrix[i][j] + distance_matrix[j][k] < distance_matrix[i][k]:
                    heuristics_matrix[i] = float('-inf')
                    break
    
    return heuristics_matrix
