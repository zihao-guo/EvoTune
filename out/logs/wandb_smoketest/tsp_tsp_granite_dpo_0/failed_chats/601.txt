import numpy as np

def heuristics(distance_matrix):
    # Add a row of ones for self-distance, to consider the edge from a vertex to itself
    distance_matrix = np.vstack([np.ones(distance_matrix.shape[1]), distance_matrix])
    
    # Compute the sum of distances for each node
    row_sums = np.sum(distance_matrix, axis=1)
    
    # Introduce an "importance" factor, based on the number of connections each node has
    importance_factor = 1 / row_sums
    importance_factor[0] = 0  # Exclude self-distance from importance calculation
    importance_factor = importance_factor / np.sum(importance_factor)  # Normalize
    
    # Compute the "least-connectedness" heuristic
    heuristics_matrix = (1 / importance_factor) * (1 - distance_matrix)
    
    return heuristics_matrix
