    n = len(distance_matrix)

    rank = np.zeros(n, dtype=int)

        rootX = find(x)
        rootY = find(y)
        if rootX == rootY:
            return

            parent[rootX] = rootY
        elif rank[rootX] > rank[rootY]:
            parent[rootY] = rootX
        else:
            parent[rootY] = rootX

        if min_idx!= -1:

    # Calculate Primal Heuristic score
