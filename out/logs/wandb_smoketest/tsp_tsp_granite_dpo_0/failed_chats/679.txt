import numpy as np
import random

def heuristics(distance_matrix, tabu_list=None, temperature=1000, alpha=0.9, max_iterations=1000):
    """
    This function computes a heuristic matrix indicating the undesirability of including each edge in a solution.

    Parameters:
    - distance_matrix (numpy array): A square matrix of non-negative distances between nodes.
    - tabu_list (list, optional): List storing indices of recently visited nodes (for tabu list). If None, no tabu list is used.
    - temperature (float, optional): Current temperature in the simulated annealing process (default is 1000).
    - alpha (float, optional): Cooling rate for the simulated annealing process (default is 0.9).
    - max_iterations (int, optional): Maximum number of iterations for the simulated annealing process (default is 1000).

    Returns:
    - numpy array: A matrix of the same shape as the input, indicating the undesirability of including each edge.
    """

    # Initialize the heuristic matrix
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    # Initialize the current node and its path cost
    current_node = 0
    current_path_cost = 0
    visited_nodes = set()

    # Run simulated annealing
    for _ in range(max_iterations):
        # Generate a neighbor
        neighbors = [node for node in range(len(distance_matrix)) if (node, current_node) in distance_matrix.indices and (current_node, node) not in visited_nodes]
        if neighbors:
            neighbor_node = random.choice(neighbors)
            neighbor_cost = distance_matrix[current_node, neighbor_node]

            # Check tabu list
            if tabu_list is not None and neighbor_node in tabu_list:
                continue

            # Calculate potential gain and accept/reject neighbor based on simulated annealing
            gain = neighbor_cost - current_path_cost
            if gain > 0 or np.exp(gain / temperature) > random.random():
                current_node = neighbor_node
                current_path_cost = neighbor_cost

                # Add current node to the tabu list (if tabu list is used)
                if tabu_list is not None:
                    tabu_list.remove(current_node)

                # Mark current node as visited
                visited_nodes.add(current_node)

        # Update heuristic matrix with the cost of the current path
        heuristic_matrix[current_node, current_node] = current_path_cost

        # Reduce temperature
        temperature *= alpha

    return heuristic_matrix

The tabu list is used to avoid cycling back to previously visited nodes. This should result in a higher score compared to a simple re-scaling of the distance matrix.
