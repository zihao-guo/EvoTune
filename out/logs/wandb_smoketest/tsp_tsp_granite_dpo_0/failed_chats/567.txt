To create a more efficient heuristics function, we'll adopt a strategy known as "Minimum Spanning Tree (MST) with Edge Weights". The idea is to calculate an estimate of how "undesirable" each edge is based on its position in the MST. The closer an edge is to the MST, the more "desirable" it is, and vice versa.

For this, we'll use Prim's algorithm to construct the MST. Then, we'll assign a weight to each edge in the distance_matrix, considering its position in the MST. Edges not in the MST will receive a higher weight, making them less desirable.

import heapq

def heuristics(distance_matrix):
    n = len(distance_matrix)

    # Initialize the MST
    mst_set = set()
    mst_edges = []
    total_weight = 0
    previous = {}

    # Prim's algorithm to find MST
    for i in range(n):
        min_distance = float('inf')
        min_index = -1

        for j in range(n):
            if (j not in mst_set) and (distance_matrix[i][j] < min_distance):
                min_distance = distance_matrix[i][j]
                min_index = j

        heapq.heappush(mst_edges, (min_distance, i, min_index))
        total_weight += min_distance
        previous[min_index] = i

    # Now, we build the MST and create heuristics for each edge
    mst = []
    while mst_edges:
        d, u, v = heapq.heappop(mst_edges)
        mst.append((u, v))

        if u in previous and v in previous[u]:
            # v is already in the MST
            continue

        # v is not in the MST, calculate heuristic
        edge_weight = distance_matrix[previous[u]][v]
        undesirability_factor = max(total_weight - distance_matrix[u][v], 0) / total_weight
        heuristic = edge_weight + undesirability_factor

        # Add to heuristics matrix
        for i in range(n):
            if (u, i) in mst_set or (i, v) in mst_set:
                distance_matrix[u][i] = max(distance_matrix[u][i], heuristic)
                distance_matrix[i][u] = heuristic

    return distance_matrix
