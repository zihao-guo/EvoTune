import numpy as np

def heuristics(distance_matrix):

    N = distance_matrix.shape[0]

    heuristic_values = np.zeros_like(distance_matrix)

    # Calculate Expected Shortest Path (ESP) for each node
    for i in range(N):
        # Current node
        curr_node = i
        
        # Initialize the set of outgoing nodes for this node
        outgoing_nodes = set(range(N))
        outgoing_nodes.remove(curr_node)

        # Initialize list to hold future distances
        future_distances = [0] * N

        # Calculate ESP
        for _ in range(10):  # Adjust this number for accuracy-performance trade-off
            # Generate future potential edge costs
            future_edge_costs = np.random.exponential(scale=1, size=len(outgoing_nodes))
            
            # Update future distances considering both current and potential future costs
            future_distances = [distance_matrix[curr_node, j] + min(future_edge_costs) for j in outgoing_nodes]

            # Sort future distances to form a priority queue
            future_distances.sort()

            # Update the heuristic value for each outgoing node
            for j in outgoing_nodes:
                heuristic_values[curr_node, j] = distance_matrix[curr_node, j] + future_distances[0]

                # Remove the chosen edge from the future distances
                future_distances[0] += distance_matrix[curr_node, j]
                outgoing_nodes.remove(j)

    return heuristic_values
