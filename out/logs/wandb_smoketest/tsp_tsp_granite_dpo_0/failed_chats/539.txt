    G = nx.from_numpy_array(distance_matrix)

    edge_betweenness = nx.edge_betweenness_centrality(G, normalized=False, weight='weight')
    
    # Convert betweenness centrality to heuristics matrix
    heuristics_matrix = np.array(list(edge_betweenness.values()))
    
    return heuristics_matrix
