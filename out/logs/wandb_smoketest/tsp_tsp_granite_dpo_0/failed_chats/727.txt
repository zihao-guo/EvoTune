import numpy as np
from scipy.sparse.csgraph import csgraph_from_dense
from scipy.spatial.distance import pdist, squareform

def heuristics(distance_matrix):
    # Create a graph from the distance matrix
    graph = squareform(pdist(distance_matrix, 'premac'))
    g = csgraph_from_dense(graph)

    # Calculate betweenness centrality for each node
    betw_centrality = np.zeros(distance_matrix.shape)
    for _ in range(distance_matrix.shape[0]):
        betweenness, _ = scipy.spatial.distance.pdist_betweenness(g, distance_matrix)
        betw_centrality += betweenness

    # Normalize betweenness centrality values between 0 and 1
    betw_centrality = betw_centrality / betw_centrality.max()

    # Create a new matrix where each entry represents the heuristic score for including an edge
    heuristics_matrix = np.outer(betw_centrality, betw_centrality)

    return heuristics_matrix
