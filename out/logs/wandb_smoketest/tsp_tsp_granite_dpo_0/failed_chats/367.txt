import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics(distance_matrix):
    # Compute the pairwise distance matrix
    D = squareform(pdist(distance_matrix))

    # Create a graph from the distance matrix
    graph = scipy.sparse.csgraph.create_adjacency_matrix(D, connectivity=1)

    # Find the MST
    MST = minimum_spanning_tree(graph)

    # Initialize heuristics matrix
    heuristics_matrix = np.copy(D)

    # Iterate over each node (except the last)
    for i in range(MST.shape[0] - 1):
        # Find nearest neighbor for current node
        nearest_neighbor_index = np.argmin(D[i, MST[i].indices])

        # Replace the current edge with its nearest neighbor
        heuristics_matrix[i, nearest_neighbor_index] = D[i, nearest_neighbor_index]
        heuristics_matrix[nearest_neighbor_index, i] = D[nearest_neighbor_index, i]

        # Remove the current node from MST
        MST = minimum_spanning_tree(graph.toarray() - np.diag(np.sum(graph.toarray(), axis=1)))
        MST[i, MST.shape[0] - 1] = np.inf

    # Update the last edge of the MST with the shortest remaining edge
    heuristics_matrix[MST.indices[np.argmin(MST.data)]] = D[MST.indices[np.argmin(MST.data)], MST.indptr[-1]]

    return heuristics_matrix
