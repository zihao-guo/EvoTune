```python
import numpy as np
from scipy.sparse.csgraph import laplacian

def heuristics(distance_matrix):
    # Convert the distance_matrix into a undirected graph (adjacency matrix)
    graph = np.abs(distance_matrix - distance_matrix.T)  # Use the Frobenius norm for symmetric matrix

    # Compute the degree centrality using Laplacian matrix
    # Laplacian matrix is related to the graph's conductance
    laplacian_matrix = laplacian(csgraph=csgraph.binarize(graph))
    degrees = np.array(np.sum(graph, axis=0)).reshape(-1)
    eigenvectors, _ = np.linalg.eig(laplacian_matrix.tocsc())
    centrality = np.array(eigenvectors).real[0] / np.sqrt(degrees)

    # Normalize centrality values to a range of 0 to 1
    centrality = (centrality - np.min(centrality)) / (np.max(centrality) - np.min(centrality))

    # Add a small constant to avoid division by zero
    centrality += 1e-9

    # Return the centrality matrix
    return centrality
```
