import numpy as np

def weighted_travelling_salesman_heuristic(distance_matrix):
    # Compute row sums as initial edge weights
    row_sums = np.sum(distance_matrix, axis=1)
    
    # Add bias term to prevent trivial solution (all shortest paths)
    max_row_sum = np.max(row_sums)
    if max_row_sum == 0:
        max_row_sum = 1  # to avoid division by zero
    heuristic_matrix = row_sums + max_row_sum

    return heuristic_matrix
