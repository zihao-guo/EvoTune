```python
import heapq

def heuristics(distance_matrix):
    # Graph representation as an adjacency list
    n = len(distance_matrix)
    graph = [[] for _ in range(n)]
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i][j]!= 0:
                graph[i].append((distance_matrix[i][j], j))
                graph[j].append((distance_matrix[j][i], i))

    # Set to keep track of included vertices
    included = set()

    # Priority queue for Prim's algorithm
    priority_queue = [(0, 0)]  # (total_weight, root_vertex)

    # Initialize the total weight to 0 and the first vertex included
    total_weight = 0
    root_vertex = 0

    # Build the MST
    while priority_queue:
        weight, current_vertex = heapq.heappop(priority_queue)
        total_weight += weight  # Add the weight of the edge to the total

        # If the vertex is already included, skip it
        if current_vertex in included:
            continue

        # Add the current vertex to the included set
        included.add(current_vertex)

        # Add neighbors to the priority queue
        for neighbor_weight, neighbor in graph[current_vertex]:
            if neighbor not in included:
                heapq.heappush(priority_queue, (neighbor_weight, neighbor))

    return total_weight
```
