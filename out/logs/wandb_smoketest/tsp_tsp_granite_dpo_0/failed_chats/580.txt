    # Dijkstra's algorithm
    while queue:
        current_cost, current_vertex = heapq.heappop(queue)

        if current_cost > costs[current_vertex]:
            continue  # Already visited, skip

            # Cost adjustment: more common edges get a discount
            adjusted_cost = current_cost + (1 - min(costs[next_vertex], current_cost)) * 0.5
            
            # Only update if the new path is shorter
            if adjusted_cost < costs[next_vertex]:
                costs[next_vertex] = adjusted_cost
                heapq.heappush(queue, (adjusted_cost, next_vertex))
