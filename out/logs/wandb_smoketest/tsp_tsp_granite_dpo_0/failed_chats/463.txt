import numpy as np

def betweenness_centrality(adj_matrix):
    n = adj_matrix.shape[0]
    path_counts = np.ones((n, n))  # Initialize with 1s
    short_paths = np.zeros((n, n))  # Initialize as zero

    for _ in range(1, n):
        for i in range(n):
            for j in range(n):
                if adj_matrix[i, j] > 0:
                    u, v = np.where(adj_matrix[i] == 1)
                    if len(u) > 0 and len(v) > 0:
                        path_counts[i, j] += short_paths[v[0], u[0]]
                        short_paths[i, j] = min(short_paths[i, j], path_counts[i, j])

    betw_centrality = np.sum(path_counts) / np.sum(short_paths)
    return betw_centrality

def heuristics(distance_matrix):

    direct_score = distance_matrix.copy()

    heuristics_score = direct_score + indirect_score + betw_centrality_matrix
