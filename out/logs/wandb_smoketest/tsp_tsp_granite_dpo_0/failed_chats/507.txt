Betweenness centrality is a measure in graph theory that indicates the number of shortest paths that pass through a given vertex (or edge in this case). The higher the betweenness centrality of an edge, the more critical it is to the connectivity of the graph.

```python
import numpy as np
from scipy.sparse.csgraph import csgraph_betweenness

def heuristics(distance_matrix):
    # Convert the distance matrix into a sparse adjacency matrix
    adjacency_matrix = csgraph_betweenness(distance_matrix, symmetric_paths=True, directed=False)
    
    # Calculate the betweenness centrality scores
    betweenness_centrality = csgraph_betweenness(adjacency_matrix, symmetric_paths=True, directed=False)[0]
    
    # Normalize the scores to be between 0 and 1
    max_centrality = np.max(betweenness_centrality)
    normalized_centrality = betweenness_centrality / max_centrality
    
    # Create the heuristics matrix
    heuristics_matrix = np.array(normalized_centrality).reshape(distance_matrix.shape)
    
    return heuristics_matrix
```
