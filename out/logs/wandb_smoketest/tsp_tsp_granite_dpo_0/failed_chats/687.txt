def heuristics(distance_matrix):
    n = len(distance_matrix)
    nad_matrix = [[0 for _ in range(n)] for _ in range(n)]

    # Iterate through each node
    for i in range(n):
        for j in range(n):
            # Check only nodes that are not the same as i or j
            if i!= j:
                # Calculate the average distance to neighboring nodes
                num_neighbors = 0
                sum_distance = 0
                for k in range(n):
                    if k!= i and k!= j:
                        sum_distance += distance_matrix[i][k]
                        num_neighbors += 1
                nad_matrix[i][j] = nad_matrix[j][i] = sum_distance / num_neighbors if num_neighbors > 0 else 0  # Avoid division by zero

    return nad_matrix
