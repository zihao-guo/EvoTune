```python
    degrees = np.sum(distance_matrix, axis=0)

    heuristic_matrix = np.zeros_like(distance_matrix)

    # Assign lower heuristic values to edges between nodes with high degrees
    for i in range(distance_matrix.shape[0]):
        for j in range(distance_matrix.shape[1]):
            if i < j:
                heuristic_matrix[i, j] = degrees[i] + degrees[j]  # High degree nodes' connections
                heuristic_matrix[j, i] = degrees[i] + degrees[j]  # Symmetry
    
    return heuristic_matrix
```
