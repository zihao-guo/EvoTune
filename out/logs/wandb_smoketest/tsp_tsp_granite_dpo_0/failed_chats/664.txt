```python
def heuristics(distance_matrix):
    # Calculate graph density (average degree)
    n = len(distance_matrix)
    avg_degree = sum(sum(distance_matrix[i]) for i in range(n)) / (n * (n - 1))
    
    # Calculate individual edge heuristic (cost / density)
    heuristics_matrix = [
        [min(1.0, d / avg_degree) for d in dist] for dist in distance_matrix
    ]
    
    return heuristics_matrix
```
