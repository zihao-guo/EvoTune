import numpy as np

def heuristics(distance_matrix):
    # Degree centrality
    degree_centrality = np.sum(distance_matrix, axis=0)
    
    # Clustering coefficient
    n = distance_matrix.shape[0]
    clustering_coefficient = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j]!= 0:
                clustering_coefficient[i, j] = 2 * distance_matrix[i, j] / np.sum(distance_matrix[i, :])
                clustering_coefficient[j, i] = clustering_coefficient[i, j]
    
    # Eigenvector centrality
    _, _ = np.linalg.eig(np.dot(distance_matrix, distance_matrix.T))
    eigenvector_centrality = np.linalg.inv(np.eye(n) - np.dot(distance_matrix, distance_matrix.T)).sum(axis=1)
    
    # Combine using weighted sum
    score_matrix = 0.3 * degree_centrality + 0.4 * clustering_coefficient + 0.3 * eigenvector_centrality
    
    return score_matrix
