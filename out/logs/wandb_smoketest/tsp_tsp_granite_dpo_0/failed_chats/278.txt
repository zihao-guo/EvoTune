import numpy as np

def heuristics(distance_matrix, future_cost_matrix, alpha=0.5, beta=0.5):
    """

    and an estimate of future costs.

    :param distance_matrix: numpy array of shape (n, n), where n is the number of nodes.
    :param future_cost_matrix: numpy array of shape (n, n), where n is the number of nodes.
    :param alpha: A hyperparameter (0 <= alpha <= 1) to balance the influence of distance vs. future cost.
    :param beta: A hyperparameter (0 <= beta <= 1) to balance the influence of future cost.

    :return: numpy array of shape (n, n), where each element represents the heuristic score of including an edge.
    """
    n = len(distance_matrix)

    # Initialize the heuristics matrix with zeros
    heuristics_matrix = np.zeros(distance_matrix.shape)

    # Iterate over each node (except the last one)
    for i in range(n - 1):
        for j in range(i + 1, n):
            # Calculate the weighted sum
            heuristics_matrix[i, j] = (alpha * distance_matrix[i, j] + beta * future_cost_matrix[i, j])

    return heuristics_matrix
