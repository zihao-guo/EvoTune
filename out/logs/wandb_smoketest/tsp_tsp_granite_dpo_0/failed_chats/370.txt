def kruskal_mst(distance_matrix):
    """
    Kruskal's Algorithm for Minimum Spanning Tree
    """
    def find(parent, i):
        if parent[i] == i:
            return i
        return find(parent, parent[i])

    def union(parent, rank, x, y):
        xroot = find(parent, x)
        yroot = find(parent, y)
        if rank[xroot] < rank[yroot]:
            parent[xroot] = yroot
        elif rank[xroot] > rank[yroot]:
            parent[yroot] = xroot
        else:
            parent[yroot] = xroot
            rank[xroot] += 1

    n = len(distance_matrix)
    parent = list(range(n))
    rank = [0] * n
    edges = sorted((dist, i, j) for i, row in enumerate(distance_matrix) for j, dist in enumerate(row))

    mst_edges = []
    for dist, i, j in edges:
        x, y = find(parent, i), find(parent, j)
        if x!= y:
            union(parent, rank, x, y)
            mst_edges.append((i, j, dist))

    return mst_edges

def heuristics(distance_matrix):
    """
    This function creates a heuristic score for each edge in the distance matrix

    mst_edges = kruskal_mst(distance_matrix)
    heuristic_matrix = [[0] * len(distance_matrix[0]) for _ in range(len(distance_matrix))]
