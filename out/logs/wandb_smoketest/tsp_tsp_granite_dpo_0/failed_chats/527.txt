```python
def heuristics(distance_matrix):
    # Get the number of nodes (n)
    n = distance_matrix.shape[0]
    
    # Initialize the heuristic matrix with zeros, same shape as distance_matrix
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    # Calculate expected cost change for each edge
    for i in range(n):
        for j in range(i+1, n):  # Consider only edges between distinct nodes to avoid redundancy
            # Calculate the maximum potential saving by excluding this edge
            max_save = distance_matrix[i, j]
            for k in range(n):
                if k!= i and k!= j:  # Avoid including node i or j in the potential saving calculation
                    max_save = max(max_save, distance_matrix[i, k] + distance_matrix[k, j] - distance_matrix[i, j])
            
            # Assign heuristic value as the maximum potential saving
            heuristic_matrix[i, j] = max_save
    
    return heuristic_matrix
```
