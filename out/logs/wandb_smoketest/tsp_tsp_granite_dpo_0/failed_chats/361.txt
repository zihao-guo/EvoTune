    n = len(distance_matrix)
    heuristic_matrix = [[0]*n for _ in range(n)]

    # Calculate the heuristic values using triangle inequality
    for i in range(n):
        for j in range(n):
            for k in range(n):
                if i!= j and j!= k and i!= k:
                    heuristic_matrix[i][j] = min(heuristic_matrix[i][j], distance_matrix[i][k] + distance_matrix[k][j])
    
    return heuristic_matrix
