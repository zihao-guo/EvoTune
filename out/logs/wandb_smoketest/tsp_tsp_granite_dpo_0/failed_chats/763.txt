```python
def heuristics(distance_matrix):
    n = len(distance_matrix)
    heuristic_matrix = [[0] * n for _ in range(n)]
    
    # Calculating the heuristic score for each node (except source node)
    for i in range(1, n):
        for j in range(n):
            if i == j:  # No-self-loop
                heuristic_matrix[i][j] = distance_matrix[i][j]
            else:
                heuristic_matrix[i][j] = distance_matrix[i][j] + min(heuristic_matrix[i][k] for k in range(n) if k!= j)  # Min-path sum heuristic
    
    return heuristic_matrix
```
