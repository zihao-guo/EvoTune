def heuristics(distance_matrix):

    path_weights = np.zeros((n, n))  # initialize the path weights matrix

    # Compute the path weights from each vertex to all others
    for i in range(n):
        for j in range(n):
            for k in range(n):
                if i!= k and j!= k:
                    path_weights[i][j] += distance_matrix[i][k] + distance_matrix[k][j]  # add edge weight to total path weight

    # Normalize the path weights to make them comparable
    path_weights = np.minimum(path_weights, distance_matrix)  # prevent unrealistically high weights

    # The heuristics score for each edge is the normalized path weight through that edge

    return heuristics_matrix
