def heuristics(distance_matrix):

    edges = np.array(np.ndindex(distance_matrix.shape)).reshape(-1, 2)

    forest = list(range(n))

    for edge in edges:
        u, v = edge
        if len(set([u, v])) < 2:  # If the edge connects the same vertex
            continue
        
        # U and V are in different trees
        if not (forest[u] == forest[v]):
            # Find the forest containing the smaller element
            smaller = min(forest[u], forest[v])
            
            # Add the edge to the result matrix (heuristics_matrix)
            heuristics_matrix[u, v] = distance_matrix[u, v]

            forest[forest == smaller] = max(u, v)
    
    # If the number of edges is not m, we need to find the shortest cycle
    cycle_length = m - len(forest)
