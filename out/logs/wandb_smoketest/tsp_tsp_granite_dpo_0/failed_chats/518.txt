import heapq

def heuristics(distance_matrix):
    n = len(distance_matrix)
    # Priority queue to store (cost, edge) tuples. Cost = (weight + factor).
    priority_queue = [(distance_matrix[i][j] + factor, (i, j)) for i in range(n) for j in range(i+1, n)]
    heapq.heapify(priority_queue)  # Convert list to a heap for efficient extraction of minimum.

    # To keep track of the best edge in terms of total cost
    best_edge = (0, None)

    # Process each edge
    while priority_queue:
        cost, (u, v) = heapq.heappop(priority_queue)

        # If the current edge is better than the previously known best, update the best.
        if cost < best_edge[0]:
            best_edge = (cost, (u, v))

        # Push the opposite edge onto the priority queue.
        if u!= v:
            cost = distance_matrix[u][v] + factor
            heapq.heappush(priority_queue, (cost, (v, u)))

    # The best_edge tuple contains the minimum total cost and the corresponding edge.
    # To get the heuristic matrix, we divide this total cost by 2 (assuming each edge is bidirectional).
    heuristic_matrix = [[best_edge[0] / 2] * n for _ in range(n)]
    return heuristic_matrix
