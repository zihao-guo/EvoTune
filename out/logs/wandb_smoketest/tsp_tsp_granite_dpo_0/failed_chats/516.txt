import numpy as np

def centrality_score(adj_matrix):
    # Calculate the adjacency matrix
    n = adj_matrix.shape[0]
    centrality = np.zeros((n, n))

    # Calculate shortest path distances using Floyd-Warshall algorithm
    dist = adj_matrix.copy()
    for k in range(n):
        for i in range(n):
            for j in range(n):
                dist[i, j] = min(dist[i, j], dist[i, k] + dist[k, j])

    # Calculate centrality scores (sum of distances from each node to all other nodes)
    for i in range(n):
        centrality[i, i] = 0  # Ignore self-loops
        for j in range(n):
            if i!= j:
                centrality[i, j] = dist[i, j]

    return centrality

def heuristics(distance_matrix):
    # Step 1: Centrality calculation
    adj_matrix = np.where(distance_matrix > 0, 1, 0)  # For undirected graphs
    centrality = centrality_score(adj_matrix)
    
    # Step 2: Convert centrality to undesirability matrix
    undesirability = np.maximum(distance_matrix - centrality, 0)
    
    return undesirability

# Note: This function assumes the input is a distance matrix (adjacency matrix with self-distances as 0).
