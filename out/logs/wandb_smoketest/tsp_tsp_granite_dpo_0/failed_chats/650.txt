This method aims to find a subset of edges in the graph such that the total weight of the edges in the subset is minimized and it forms a tree that spans all vertices. import heapq

def mst_heuristics(distance_matrix):
    num_vertices = len(distance_matrix)
    
    # Initialize the minimum edge heaps with (weight, source, destination) pairs
    heaps = [[distance_matrix[i][j], i, j] for i in range(num_vertices) for j in range(i+1, num_vertices)]
    heapq.heapify(heaps)
    
    # Initialize the result matrix with zeros
    result = [[0] * num_vertices for _ in range(num_vertices)]
    
    # Process edges in increasing order of weight
    while heaps:
        # Get the edge with minimum weight
        weight, u, v = heapq.heappop(heaps)
        
        # If both endpoints are included in the tree, there's no need to add this edge
        if result[u][v]:
            continue
        
        # Include this edge in the tree and update the result matrix
        result[u][v] = result[v][u] = 1
        # Remove the other endpoint from the heaps
        if u!= v:
            heapq.heappush(heaps, [distance_matrix[u][w], u, w] for w in range(num_vertices) if w!= v)
    
    return result
