def heuristics(distance_matrix):

    n = len(distance_matrix)

    # Calculate impact for each edge
    for i in range(n):

            impact = distance_matrix[i][j]
            for k in range(n):
                if i!= k and j!= k:  # Avoid self-loop and diagonal
                    impact -= distance_matrix[i][k] * distance_matrix[k][j]
            impact_matrix[i][j] = impact  # Store impact in matrix

    return impact_matrix
