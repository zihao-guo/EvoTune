def heuristics(distance_matrix):

    cost_to_complete = [sum(distance_matrix[i]) for i in range(n)]

    # Calculate cost-to-complete for each node
    for node in range(n):

    undesirability_matrix = [[(cost_to_complete[i] + cost_to_complete[j] - distance_matrix[i][j]) / 2 for j in range(n)] for i in range(n)]
