import numpy as np

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]  # number of nodes
    avg_dist = np.mean(distance_matrix, axis=0)  # average distance from each node
    path_counts = np.sum(distance_matrix, axis=1)  # total number of paths from each node
    dand = np.mean(distance_matrix[np.arange(n), :] * np.arange(n).reshape(-1, 1), axis=1)  # DAND
    weighted_degree = np.sum(distance_matrix, axis=0)  # weighted degree for each node

    # Combine these metrics into a single score matrix
    heuristics_matrix = (1 - avg_dist) * path_counts + (avg_dist - dand) * weighted_degree

    # Normalize the scores so they sum to 1
    heuristics_matrix /= np.sum(heuristics_matrix, axis=0)

    return heuristics_matrix
