import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from numpy.random import randint

def heuristics(distance_matrix):
    # Step 1: Build Minimum Spanning Tree (MST)
    mst = minimum_spanning_tree(csgraph=distance_matrix)
    mst_edges = np.argwhere(mst)

    # Step 2: Calculate nearest neighbor heuristic
    nn_heuristic = np.zeros_like(distance_matrix, dtype=float)
    for i in range(distance_matrix.shape[0]):
        nn_indices = np.argmin(distance_matrix[i, :])
        nn_heuristic[i, nn_indices] = distance_matrix[i, nn_indices]
        nn_heuristic[nn_indices, i] = distance_matrix[nn_indices, i]

    # Step 3: Combine heuristics
    heuristics_matrix = nn_heuristic + distance_matrix - mst_edges[:, 0] * mst_edges[:, 1]

    return heuristics_matrix
