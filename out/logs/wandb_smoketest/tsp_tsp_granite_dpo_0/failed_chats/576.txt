import numpy as np

def weights(distance_matrix):
    """Assign weights to nodes based on their positions or importance."""
    # Assuming nodes are in a 2D plane
    weights = np.exp(-distance_matrix**2)  # Here, we use an exponential decay function to give more importance to nearby nodes
    return weights / np.sum(weights)  # Normalize so that weights sum up to 1

def cost_to_include(distance_matrix, weights):
    """Calculate the cost to include each edge in a solution."""
    n = distance_matrix.shape[0]
    cost_matrix = np.zeros_like(distance_matrix)

    # Iteratively calculate cost_to_include for each node
    for i in range(n):
        for j in range(n):
            if i == j:  # Self-loops have zero cost
                cost_matrix[i, j] = 0
                continue

            # Consider the impact of both the current node and its neighbors
            total_cost = 0
            for k in range(n):
                # Calculate cost based on the distance to the current node and the importance of its neighbors
                total_cost += distance_matrix[i, k] * weights[k, j]

            cost_matrix[i, j] = total_cost

    return cost_matrix

def heuristics(distance_matrix):
    """Compute a cost matrix where the value at position (i, j) is a heuristic estimate of undesirability to include edge (i, j)."""
    weights = weights(distance_matrix)
    return cost_to_include(distance_matrix, weights)
