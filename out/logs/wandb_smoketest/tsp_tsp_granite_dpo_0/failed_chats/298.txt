import numpy as np

def betweenness_centrality(graph):
    # Graph representation can be adjacency matrix or adjacency list
    # For simplicity, let's assume we have an adjacency matrix 'graph'
    n = graph.shape[0]
    betweenness = np.zeros((n, n))
    
    for node in range(n):
        # Compute shortest paths between all pairs of nodes
        paths = _dijkstra(graph, node)
        
        # Number of shortest paths that pass through node 'node'
        num_paths = np.sum(paths[:, node] > 0)
        
        # Betweenness for node 'node'
        for i in range(n):
            if i!= node:
                betweenness[i, node] = np.sum(paths[node, i] > 0) / num_paths
                betweenness[node, i] = betweenness[i, node]
    
    return betweenness

def _dijkstra(graph, start):
    n = graph.shape[0]
    dist = np.zeros(n)
    prev = np.zeros((n, n), dtype=int)
    visited = np.zeros(n, dtype=bool)
    
    dist[start] = 0
    visited[start] = True
    
    for _ in range(n - 1):
        min_dist = float('inf')
        u = -1
        for v in range(n):
            if not visited[v] and dist[v] < min_dist:
                min_dist = dist[v]
                u = v
        
        visited[u] = True
        
        for v in range(n):
            if graph[u, v] > 0 and not visited[v]:
                alt = dist[u] + graph[u, v]
                if alt < dist[v]:
                    dist[v] = alt
                    prev[v][u] = 1
                    
    return dist, prev

def heuristics(distance_matrix):
