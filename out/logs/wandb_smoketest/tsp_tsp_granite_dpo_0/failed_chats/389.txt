```python
def heuristics(distance_matrix):
    # Get the shape of the distance matrix
    num_nodes = distance_matrix.shape[0]
    
    # Initialize the heuristic matrix with the same size as the input
    heuristic_matrix = distance_matrix.copy()
    
    # Run bidirectional search
    def bidirectional_search(start, goal):
        queue = [(0, start)]  # (cost, node)
        visited = set([start])
        while queue:
            (cost, current) = queue.pop(0)
            if current == goal:
                return cost
            
            # Explore along the path from current to goal
            for neighbor, edge_cost in enumerate(distance_matrix[current]):
                if neighbor not in visited:
                    visited.add(neighbor)
                    new_cost = cost + edge_cost
                    queue.append((new_cost, neighbor))
            
            # Explore from the goal towards the start
            reverse_queue = [(cost, goal) for (cost, node) in enumerate(distance_matrix[::-1])]
            while reverse_queue:
                (reverse_cost, goal) = reverse_queue.pop(0)
                if goal not in visited:
                    visited.add(goal)
                    new_cost = reverse_cost + distance_matrix[goal][start]
                    heuristic_matrix[goal][start] = new_cost
                    queue.append((new_cost, start))
                
                for neighbor, edge_cost in enumerate(distance_matrix[goal]):
                    if neighbor not in visited:
                        visited.add(neighbor)
                        new_cost = reverse_cost + edge_cost
                        heuristic_matrix[goal][neighbor] = new_cost
                        queue.append((new_cost, neighbor))
        
        return None  # No path found

    # Perform bidirectional search for all nodes
    for i in range(num_nodes):
        bidirectional_search(i, (i+1) % num_nodes)

    return heuristic_matrix
```

This heuristics() function performs a bidirectional search, estimating the cost from each node to all other nodes, and from all other nodes to each node. The resulting heuristic matrix will have higher values where direct paths are present (and thus more undesirable) and lower values for longer or indirect routes. The use of two independent searches increases the reliability of the cost estimate compared to a simple unidirectional search.
