import numpy as np

def heuristics(distance_matrix):
    # Reciprocity: For every edge (i, j) in the matrix, add the weight of (j, i)
    reciprocity_score = distance_matrix / distance_matrix
    
    # Minimum Weight Spanning Tree (MWST) heuristic
    def min_weight_spanning_tree(adj_matrix):
        n = len(adj_matrix)
        parent = list(range(n))
        rank = [0] * n
        
        def find(x):
            if parent[x]!= x:
                parent[x] = find(parent[x])
            return parent[x]

        def union(x, y):
            root_x = find(x)
            root_y = find(y)
            if root_x == root_y:
                return
            if rank[root_x] > rank[root_y]:
                parent[root_y] = root_x
            else:
                parent[root_x] = root_y
                if rank[root_x] == rank[root_y]:
                    rank[root_y] += 1
        
        for edge in enumerate(adj_matrix):
            if edge[1] not in parent[edge[0]]:
                union(*edge)
        
        return adj_matrix[edge[0]][edge[1]] for edge in enumerate(adj_matrix) if find(edge[0])!= find(edge[1])
    
    # Bellman-Held-Karp lower bound
    def lower_bound(dist_matrix):
        n = len(dist_matrix)
        L = np.inf
        V = [float('inf')] * n
        V[0] = 0
        for _ in range(n):
            for u in range(n):
                for v in range(u+1, n):
                    L = min(L, dist_matrix[u][v] + V[u] + V[v])
                    if L < dist_matrix[u][v]:
                        break  # No need to check further edges for u and v

                    break  # No need to check further vertices for u
            else:
                break  # No need to check further vertices for u
            
        return L
    
    # Combine the three scores
    undesirability_score = reciprocity_score - min_weight_spanning_tree(distance_matrix) + lower_bound(distance_matrix)
    
    return undesirability_score
