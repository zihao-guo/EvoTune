def heuristics(distance_matrix):
    import numpy as np

    num_vertices = distance_matrix.shape[0]
    edge_weights = distance_matrix.flatten()
    vertex_degrees = np.sum(distance_matrix, axis=1)
    avg_edge_weight_per_vertex = edge_weights / (num_vertices - 1)  # Subtract 1 because we're considering each vertex as a vertex-target pair

    heuristics_matrix = -edge_weights  # Initialize as -1 for each edge (to be adjusted later)

    for i in range(num_vertices):
        for j in range(i+1, num_vertices):  # Only compare vertices that are not the same
            heuristics_matrix[i*num_vertices + j, 0] = -edge_weights[i*num_vertices + j]  # Set weight
            heuristics_matrix[i*num_vertices + j, 1] = (vertex_degrees[i] + 1) / avg_edge_weight_per_vertex[i]  # Degree and average weight
            heuristics_matrix[j*num_vertices + i, 0] = heuristics_matrix[i*num_vertices + j, 0]  # Symmetry

    return heuristics_matrix
