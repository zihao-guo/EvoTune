import numpy as np

def page_rank(distance_matrix, alpha=0.85, damping_factor=0.85, max_iter=100, tolerance=1e-8):
    """
    Calculate Google PageRank for a given distance matrix.
    Alpha and damping factor are parameters for the PageRank algorithm.
    max_iter is the maximum number of iterations, and tolerance is the convergence criteria.
    """
    n = distance_matrix.shape[0]
    rank = np.ones(n) / n  # Initialize ranks as 1/n for each node
    
    for _ in range(max_iter):
        new_rank = np.zeros(n)
        for i in range(n):
            for j in range(n):
                if distance_matrix[i, j] > 0:  # Only consider edges with non-zero distance
                    new_rank[i] += (1 - damping_factor) * (distance_matrix[i, j] / np.sum(distance_matrix)) * rank[j]
        # Normalize ranks
        new_rank /= np.sum(new_rank)
        
        # Convergence check
        if np.linalg.norm(new_rank - rank) < tolerance:
            break
        
        rank = new_rank
    
    return rank

def heuristics(distance_matrix):

    page_ranks = page_rank(distance_matrix)
    
    # Normalize the page ranks to ensure they sum to 1, mimicking a probability distribution
    total = np.sum(page_ranks)
    return page_ranks / total
