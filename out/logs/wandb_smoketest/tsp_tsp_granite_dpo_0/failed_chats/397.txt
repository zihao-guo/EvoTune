from scipy.cluster.hierarchy import linkage, fcluster

    # Perform hierarchical clustering using Ward method
    Z = linkage(dists, method='ward')
    
    # Compute the cluster labels using fcluster with a specified depth
    cluster_labels = fcluster(Z, max_cluster_size, criterion='distance')
    
    # Create a new NxN matrix with the same shape as the input
    heuristic_matrix = np.zeros(distance_matrix.shape)
    
    # Assign lower values to edges that belong to the same cluster
    for i in range(len(distance_matrix)):
        heuristic_matrix[i, :] = cluster_labels[Z.parents[i]]

    for i in range(len(distance_matrix)):

    return heuristic_matrix
