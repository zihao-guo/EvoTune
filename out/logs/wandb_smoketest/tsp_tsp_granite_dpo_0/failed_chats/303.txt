import numpy as np

def heuristics(distance_matrix):
    # Assuming the input distance matrix is a square matrix where each element represents a Euclidean distance
    n = distance_matrix.shape[0]
    
    # Initialize the heuristic matrix with zeros
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    # Calculate the Hamming distance for each pair of vertices
    for i in range(n):
        for j in range(i+1, n):
            diff = np.abs(distance_matrix[i] - distance_matrix[j]) - 1
            heuristic_matrix[i, j] = heuristic_matrix[j, i] = diff
    
    return heuristic_matrix

# Example usage:
distance_matrix = np.array([[0, 2, 5], [2, 0, 1], [5, 1, 0]])
print(heuristics(distance_matrix))
