import heapq

def dijkstra(distance_matrix):
    n = len(distance_matrix)
    # Initialize distances from the source node (0) to all other nodes as infinite
    distances = [float('inf')] * n
    distances[0] = 0

    # Priority queue to store (distance, node) tuples
    queue = [(0, 0)]

    while queue:
        current_distance, current_node = heapq.heappop(queue)

        if current_distance > distances[current_node]:
            continue

        for neighbor, edge_weight in enumerate(distance_matrix[current_node]):
            if edge_weight > 0:
                distance = current_distance + edge_weight

                # If a shorter path to the neighbor is found, update the distance
                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    heapq.heappush(queue, (distance, neighbor))

    return distances

def heuristics(distance_matrix):
    # Calculate shortest path distances from each node to all others
    shortest_paths = dijkstra(distance_matrix)

    # Create a new matrix to store heuristic values
    heuristic_values = [[0] * len(distance_matrix[0]) for _ in range(len(distance_matrix))]

    # Populate the new matrix
    for i in range(len(distance_matrix)):
        for j in range(len(distance_matrix[0])):
            if i!= j:
                # The cost of including edge (i, j) is the sum of the shortest distances from both nodes to the other node
                cost = shortest_paths[i] + shortest_paths[j] - distance_matrix[i][j]
                heuristic_values[i][j] = cost

    return heuristic_values
