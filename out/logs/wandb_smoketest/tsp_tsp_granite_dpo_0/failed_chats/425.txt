def weighted_path_count(distance_matrix):
    num_nodes = len(distance_matrix)
    edges = []

    # Add edges with zero weight for every pair of nodes
    for i in range(num_nodes):
        for j in range(i+1, num_nodes):
            edges.append((i, j, 1))
            edges.append((j, i, 1))  # Consider bidirectional edges for path count

    # Calculate the count of paths from each node
    path_count = [[0 for _ in range(num_nodes)] for _ in range(num_nodes)]
    for edge in edges:
        u, v, weight = edge
        for k in range(num_nodes):
            if (u, k) in path_count[v] or (k, v) in path_count[u]:
                path_count[u][k] += weight
                path_count[v][k] += weight

    # Create the heuristic matrix
    heuristics = [[0 for _ in range(num_nodes)] for _ in range(num_nodes)]
    for i in range(num_nodes):
        for j in range(num_nodes):
            if i!= j:  # Paths from node i to node j are not considered
                heuristics[i][j] = distance_matrix[i][j] + path_count[i][j] / (distance_matrix[i][j] + 1)
    return heuristics
