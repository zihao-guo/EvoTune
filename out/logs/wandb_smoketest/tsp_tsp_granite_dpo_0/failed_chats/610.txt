import numpy as np

def betweenness_centrality(graph, nodes):
    n = graph.shape[0]
    degree = np.sum(graph, axis=0)
    path_length = np.diag(np.sqrt(np.sum(graph**2, axis=1)))
    trans_graph = np.eye(n) - np.linalg.inv(degree)
    b = np.zeros(n)
    
    for _ in range(2, n+1):
        new_b = np.copy(b)
        for node in nodes:
            for i in range(n):
                if i!= node:
                    parallel_edges = trans_graph[i, :].dot(graph[:, node])
                    if parallel_edges.sum() > 0:
                        b[node] += 1 / parallel_edges[i] * (path_length[i, node] - path_length[node, i])
        b = np.maximum(b, new_b)
    
    return b

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    degree = np.sum(distance_matrix, axis=0)
    avg_distance = np.mean(distance_matrix)
    connectability = betweenness_centrality(np.eye(n), np.arange(n))
    
    # Connectability score assumes lower values mean more central

    connectability = 1 / connectability
    
    # Combine average distance and connectability score
