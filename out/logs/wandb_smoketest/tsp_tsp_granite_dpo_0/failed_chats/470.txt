import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse.csgraph import diameter

def heuristics(distance_matrix):
    n = distance_matrix.shape[0]
    # Compute the adjacency matrix from distance matrix
    adj_matrix = np.pad(distance_matrix, ((1, 0), (0, 1)), mode='constant') > 0

    # Calculate the Relaxed Steinhaus Center
    dist_matrix = distance_matrix.copy()
    for i in range(n):
        for j in range(n):
            dist_matrix[i, j] = dist_matrix[i, j] if adj_matrix[i, j] else np.inf  # Prevent self-loops
        center = np.min(dist_matrix, axis=1)
        center_index = np.argmin(center)
        dist_matrix[center_index, :] = np.inf
        dist_matrix[:, center_index] = np.inf

    # Compute minimum spanning tree
    mst = minimum_spanning_tree(csgraph=dist_matrix)

    # Calculate diameter of the MST
    diameter_val = diameter(csgraph=mst)

    # Normalize diameter to get a centrality measure
    centrality_scores = np.power(diameter_val, -1)
    centrality_scores /= np.sum(centrality_scores)  # Normalize to sum up to 1

    # Combine with distance matrix
    heuristics_matrix = distance_matrix.copy()
    heuristics_matrix[np.arange(n), np.arange(n)] = centrality_scores

    return heuristics_matrix
