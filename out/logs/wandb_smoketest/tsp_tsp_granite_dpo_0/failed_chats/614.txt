import numpy as np
import random

def heuristics_NN(distance_matrix):
    num_vertices = distance_matrix.shape[0]
    # Initialize tour with a random vertex
    tour = np.random.choice(num_vertices)
    # Initialize total heuristic value
    total_heuristic = 0
    # Initialize a queue for neighbors
    neighbors_queue = []
    
    for _ in range(num_vertices):
        # Calculate potential neighbors
        potential_neighbors = [v for v in range(num_vertices) if v!= tour and distance_matrix[tour, v] < distance_matrix[tour, tour]]
        
        if len(potential_neighbors) == 0:
            # If no neighbors, perturb and choose a random vertex
            heuristic_perturbation = random.uniform(0, 1)
            if heuristic_perturbation < 0.1:  # Arbitrary threshold for perturbation
                potential_neighbors = [v for v in range(num_vertices) if v!= tour]
        
        # Choose the next vertex using nearest neighbor with perturbation
        next_vertex = potential_neighbors[np.argmin([distance_matrix[tour, v] + distance_matrix[v, np.random.randint(num_vertices)] for v in potential_neighbors])]
        
        # Add the chosen vertex to the tour and update total heuristic
        tour = next_vertex
        total_heuristic += distance_matrix[tour, next_vertex]
        
        # Add the new edge to the neighbors queue
        neighbors_queue.append((tour, next_vertex))
        
        # Move the current vertex to the end of the queue to consider its effect on the partial tour's endpoints
        neighbors_queue.append((next_vertex, tour))
    
    return total_heuristic
