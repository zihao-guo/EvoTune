def weighted_centrality(distance_matrix):
    # Get dimensions of the distance matrix
    num_nodes, _ = distance_matrix.shape
    
    # Initialize a new matrix with the same shape
    heuristics_matrix = np.copy(distance_matrix)
    
    # Iterate over each node
    for node in range(num_nodes):
        # Get neighbors of the current node
        neighbors_indices = np.where(distance_matrix[node]!= np.inf)[0]
        
        # Calculate the sum of distances to all neighbors
        distances_to_neighbors = np.sum(distance_matrix[node, neighbors_indices], axis=0)
        
        # Assign a heuristic score
        heuristics_matrix[node, neighbors_indices] = 1 / distances_to_neighbors  # Use reciprocal of distance as a heuristic
    
    return heuristics_matrix
