import networkx as nx

def heuristics(distance_matrix):

    # Find the minimum spanning tree (MST) of the graph
    mst_cost = nx.minimum_spanning_tree_algorithm(G, weight='weight')
    
    # Create a new matrix to store the heuristic values
    heuristics_matrix = np.zeros_like(distance_matrix)

    # Assign each non-edge (i,j) a heuristic value based on its position in MST
    for i in range(distance_matrix.shape[0]):
        for j in range(i+1, distance_matrix.shape[1]):
            # If (i,j) is not in the MST, its cost is the edge weight
            if (i,j) not in mst_cost.edges():
                heuristics_matrix[i,j] = distance_matrix[i,j]
            # If (i,j) is in the MST, its cost is the MST cost between its endpoints
            else:
                a, b = min(i,j), max(i,j)

    return heuristics_matrix
