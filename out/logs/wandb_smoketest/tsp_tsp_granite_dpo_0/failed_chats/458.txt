def heuristics(distance_matrix):

                weight = distance_matrix[i][j]
                cumulative_weights[i] += weight
                cumulative_weights[j] += weight

    heuristic_matrix = [[(cumulative_weights[i] + cumulative_weights[j]) / (total_weight) for j in range(num_vertices)] for i in range(num_vertices)]

    dijkstra(distance_matrix, 0)

    # Weighted random selection to get final heuristic
